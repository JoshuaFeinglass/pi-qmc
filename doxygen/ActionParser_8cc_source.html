756 NodeModel* ActionParser::parseNodeModel(const xmlXPathContextPtr& ctxt,
757  xmlNodePtr &actNode, const Species &species) {
758  NodeModel *nodeModel=0;
759  bool useHungarian=getBoolAttribute(actNode,"useHungarian");
760  int useIterations=getIntAttribute(actNode,"useIterations");
761  double nodalFactor=getDoubleAttribute(actNode,"nodalFactor");
762  if (nodalFactor<=0) nodalFactor=5.0;
763  if (useIterations<0) useIterations=0;
764  double t=getEnergyAttribute(actNode,"temperature");
765  if (t==0) t=simInfo.getTemperature();
766  std::string modelName=getStringAttribute(actNode,"model");
767  if (modelName=="SHONodes") {
768  const double omega=getEnergyAttribute(actNode,"omega");
769  nodeModel=new SHONodes(simInfo,species,omega,t,maxlevel);
770  } else if (modelName=="GSSNode" || modelName=="GroundStateSNode") {
771  std::string centerName=getStringAttribute(actNode,"center");
772  const int icenter=simInfo.getSpecies(centerName).ifirst;
773  nodeModel=new GroundStateSNode(species,icenter,simInfo.getTau());
774  } else if (modelName=="WireNodes") {
775  const double omega=getEnergyAttribute(actNode,"omega");
776  const bool updates=getBoolAttribute(actNode,"useUpdates");
777  Vec center;
778  for (int idim=0; idim<NDIM; ++idim) {
779  center[idim]=getLengthAttribute(actNode,
780  std::string(dimName).substr(idim,1));
781  }
782  int maxMovers=0;
783  if (updates) maxMovers=3;
784  nodeModel=new WireNodes(simInfo,species,omega,t,center,maxlevel,
785  updates,maxMovers);
786  } else if (modelName=="ExcitonNodes") {
787  std::string specName=getStringAttribute(actNode,"species1");
788  const Species& species1(simInfo.getSpecies(specName));
789  specName=getStringAttribute(actNode,"species2");
790  const Species& species2(simInfo.getSpecies(specName));
791  const double radius=getLengthAttribute(actNode,"radius");
792  const bool updates=getBoolAttribute(actNode,"useUpdates");
793  int maxMovers=3;
794  nodeModel=new ExcitonNodes(simInfo,species1,species2,
795  t,maxlevel,radius,updates,maxMovers);
796  } else if (modelName=="AugmentedNodes") {
797  const bool updates=getBoolAttribute(actNode,"useUpdates");
798  const bool useHungarian=getBoolAttribute(actNode,"useHungarian");
799  int maxMovers=3;
800  std::vector<const AtomicOrbitalDM*> orbitals;
801  parseOrbitalDM(orbitals, species, ctxt);
802  nodeModel=new AugmentedNodes(simInfo,species,
803  t,maxlevel,updates,maxMovers,orbitals,useHungarian);
804  } else {
805  const bool updates=getBoolAttribute(actNode,"useUpdates");
806  int maxMovers=0;
807  if (updates) maxMovers=3;
808  nodeModel=new FreeParticleNodes(simInfo,species,t,maxlevel,updates,
809  maxMovers,useHungarian,useIterations, nodalFactor);
810  }
811  return nodeModel;
812 }
