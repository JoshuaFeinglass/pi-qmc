92  const int nSlice=sectionBeads.getNSlice();
93  const int nMoving=index.size();

95  // First check for node crossing.

97  for (int i=0; i<npart; ++i) r2(i)=otherBeads(i+ifirst,islice);

99  for (int i=0; i<nMoving; ++i) r1(index(i)-ifirst)=movingBeads(i,islice);
100  newDet(islice)=evaluateUpdate(r1,r2,islice);
101  if (newDet(islice)<0) return deltaAction=2e100;
102  }
103  // Calculate the nodal action if level=0;
104  if (level==0) {
105  const double twomovert=2.0/tau; //move factor of mass in the distance^2
106  double d0(0),newd0(0),d1(0),newd1(0);
107  for (int islice=0;islice<nSlice; ++islice) {
108  *newMatrix[islice]=*matrix[islice];
109  d1=0; for (int i=0;i<npart;++i) d1+=1./(dist(islice,i)*dist(islice,i));
110  newd1=d1=sqrt(1./d1);
111  if (islice>0) {
112  for (int i=0; i<npart; ++i) r2(i)=otherBeads(i+ifirst,islice);
113  for (int i=0; i<npart; ++i) r1(i)=sectionBeads(i+ifirst,islice);
114  for (int i=0; i<nMoving; ++i) r1(index(i)-ifirst)=movingBeads(i,islice);
115  newd1=evaluateGrad(islice);
116  deltaAction+=-log(1-exp(-twomovert*newd1*newd0))
117  +log(1-exp(-twomovert*d1*d0));

119  }

121  }

191 double AnisotropicNodes::evaluate(const VArray& r1, const VArray& r2,
192  Matrix& mat) {
