81  mat=0.;
82  for(int jpart=0; jpart<npart; ++jpart) {
83  for(int ipart=0; ipart<npart; ++ipart) {
84  Vec delta(r1(jpart+ifirst)-r2(ipart+ifirst));
85  cell.pbc(delta);
86  double ear2=scale*density;
87  for (int i=0; i<NDIM; ++i) {
88  ear2 *= pg[i]->evaluate(delta[i]);
89  }

91  }
92  }
93  // Add contribution from orbitals.
94  for (std::vector<const AtomicOrbitalDM*>::const_iterator orb
95  = orbitals.begin(); orb != orbitals.end(); ++orb) {
96  int kfirst=(*orb)->ifirst, nkpart=(*orb)->npart;
97  VArray2& delta1((*orb)->getD1Array());
98  for (int jpart=0; jpart<npart; ++jpart) {
99  for (int kpart=0; kpart<nkpart; ++kpart) {
100  Vec delta(r1(jpart+ifirst)-r1(kpart+kfirst));
101  cell.pbc(delta);
102  delta1(jpart,kpart)=delta;
103  }
104  }
105  VArray2& delta2((*orb)->getD2Array());
106  for(int ipart=0; ipart<npart; ++ipart) {
107  for (int kpart=0; kpart<nkpart; ++kpart) {
108  Vec delta(r2(ipart+ifirst)-r2(kpart+kfirst));
109  cell.pbc(delta);
110  delta2(ipart,kpart)=delta;
111  }
112  }
113  (*orb)->evaluateValue(mat,scale);
114  }
115  for(int jpart=0; jpart<npart; ++jpart) {
116  for(int ipart=0; ipart<npart; ++ipart) {
117  uarray(ipart,jpart)=-log(fabs(mat(ipart,jpart))+1e-100);
118  }
119  }
120  // Find dominant contribution to determinant (destroys uarray).
121  if (useHungarian) {
122  hungarian->solve(uarray.data());
123  for (int ipart=0; ipart<npart; ++ipart) {
124  kindex(islice,ipart) = (*hungarian)[ipart];
125  }
126  // Note: u(ipart,jpart=kindex(islice,ipart)) makes maximum contribution
127  // or lowest total action.
128  }

130  int info=0;//LU decomposition

132  if (info!=0) {

135  nerror++;

138  std::exit(-1);
139  }
140  }

143  det *= mat(i,i);

152  std::cout << "too many errors!!!!" << std::endl;
153  std::exit(-1);
154  }
155  }

157  if (scaleMagnitude && (!result.err
158  && (fabs(det)<1e-50 || fabs(det)>1e50) )) {
159  if (fabs(det)<1e-250) {

161  } else {

174 void AugmentedNodes::evaluateDotDistance(const VArray &r1, const VArray &r2,
175  const int islice, Array &d1, Array &d2) {
176  std::vector<PeriodicGaussian*> pgSave(NDIM);
177  for (int i=0; i<NDIM; ++i) pgSave[i]=pg[i];
178  double tauSave=tau;
179  // Calculate gradient with finite differences.
180  // First calculate distance at smaller beta.
181  tau = tauSave*(1.-EPSILON);
182  for (int i=0; i<NDIM; ++i) pg[i]=pgm[i];
183  evaluate(r1, r2, islice, false);
184  evaluateDistance(r1, r2, islice, temp1, temp2);
185  // Next calculate distance at larger beta.
186  tau = tauSave*(1.+EPSILON);
187  for (int i=0; i<NDIM; ++i) pg[i]=pgp[i];
188  evaluate(r1, r2, islice, false);
189  evaluateDistance(r1, r2, islice, d1, d2);
190  // Now use finite differences.
191  double denom=1./(2*tau*EPSILON);
192  d1 = denom*(d1-temp1);
193  d2 = denom*(d2-temp2);
194  // Restore pg and tau to proper temperature.
195  for (int i=0; i<NDIM; ++i) pg[i]=pgSave[i];
196  tau=tauSave;

199 void AugmentedNodes::evaluateDistance(const VArray& r1, const VArray& r2,
200  const int islice, Array& d1, Array& d2) {
201  Matrix& mat(*matrix[islice]);
202  // Calculate log gradients to estimate distance.
203  d1=200; d2=200; // Initialize distances to a very large value.
204  // Set up distances for orbitals.
205  for (std::vector<const AtomicOrbitalDM*>::const_iterator orb
206  = orbitals.begin(); orb != orbitals.end(); ++orb) {
207  int kfirst=(*orb)->ifirst, nkpart=(*orb)->npart;
208  VArray2& delta1((*orb)->getD1Array());
209  for (int jpart=0; jpart<npart; ++jpart) {
210  for (int kpart=0; kpart<nkpart; ++kpart) {
211  Vec delta(r1(jpart+ifirst)-r1(kpart+kfirst));
212  cell.pbc(delta);
213  delta1(jpart,kpart)=delta;
214  }
215  }
216  VArray2& delta2((*orb)->getD2Array());
217  for(int ipart=0; ipart<npart; ++ipart) {
218  for (int kpart=0; kpart<nkpart; ++kpart) {
219  Vec delta(r2(ipart+ifirst)-r2(kpart+kfirst));
220  cell.pbc(delta);
221  delta2(ipart,kpart)=delta;
222  }
223  }
224  (*orb)->evaluateValueAndGrad();
225  }
226  // Calculate gradients for slice 1.
227  for (int jpart=0; jpart<npart; ++jpart) {
228  Vec logGrad=0.0, fgrad=0.0;
229  for (int ipart=0; ipart<npart; ++ipart) {
230  AtomicOrbitalDM::ValAndGrad orbital;
231  for (std::vector<const AtomicOrbitalDM*>::const_iterator orb
232  = orbitals.begin(); orb != orbitals.end(); ++orb) {
233  AtomicOrbitalDM::ValAndGrad temp = (**orb)(ipart,jpart);
234  orbital += temp;
235  }
236  cache(jpart,ipart) = orbital.val;
237  cacheV(jpart,ipart) = orbital.grad2;
238  Vec delta=r1(jpart+ifirst)-r2(ipart+ifirst);
239  cell.pbc(delta);
240  Vec grad; double val=density;
241  for (int i=0; i<NDIM; ++i) {
242  double value = pg[i]->evaluate(delta[i]);
243  grad[i] = pg[i]->getGradient() / (value + 1e-300);
244  val *= value;
245  }
246  grad *= val;
247  grad += orbital.grad1;
248  val += orbital.val;
249  if (useHungarian && jpart==kindex(islice,ipart)) fgrad=grad/(val+1e-300);
250  logGrad += mat(jpart,ipart)*grad*scale;
251  }
252  gradArray1(jpart) = logGrad - fgrad;
253  d1(jpart+ifirst) = sqrt(2*mass/
254  ((dot(gradArray1(jpart),gradArray1(jpart))+1e-300)*tau));
255  }
256  // Calculate gradients for slice 2.
257  for (int ipart=0; ipart<npart; ++ipart) {
258  Vec logGrad=0.0, fgrad=0.0;
259  for (int jpart=0; jpart<npart; ++jpart) {
260  Vec delta=r2(ipart+ifirst)-r1(jpart+ifirst);
261  cell.pbc(delta);
262  Vec grad; double val=density;
263  for (int i=0; i<NDIM; ++i) {
264  double value = pg[i]->evaluate(delta[i]);
265  grad[i] = pg[i]->getGradient() / (value + 1e-300);
266  val *= value;
267  }
268  grad *= val;
269  grad += cacheV(jpart,ipart);
270  val += cache(jpart,ipart);
271  if (useHungarian && jpart==kindex(islice,ipart)) fgrad=grad/(val+1e-300);
272  logGrad += mat(jpart,ipart)*grad*scale;
273  }
274  gradArray2(ipart) = logGrad - fgrad;
275  d2(ipart+ifirst) = sqrt(2*mass/
276  ((dot(gradArray2(ipart),gradArray2(ipart))+1e-300)*tau));
277  }

280 void AugmentedNodes::evaluateGradLogDist(const VArray &r1, const VArray &r2,
281  const int islice, VMatrix &gradd1, VMatrix &gradd2,
282  const Array& dist1, const Array& dist2) {
283  gradd1=0.; gradd2=0.;
284  const Matrix& mat(*matrix[islice]);
285  // ipart is the index of the particle in the gradient.
286  // jpart is the index of the particle in the distance.
287  for (int ipart=0; ipart<npart; ++ipart) {
288  for (int jpart=0; jpart<npart; ++jpart) {
289  // First treat d1 terms.
290  if (ipart==jpart) {
291  Mat d2ii;
292  d2ii = 0.0;
293  for(int kpart=0; kpart<npart; ++kpart) {
294  Vec delta=r1(ipart+ifirst)-r2(kpart+ifirst);
295  cell.pbc(delta);
296  Mat d2iik;
297  for (int i=0; i<NDIM; ++i) {
298  for (int j=0; j<NDIM; ++j) {
299  if (i==j) {
300  double value = pg[i]->evaluate(delta[i]);
301  d2iik(i,j) = pg[i]->getSecondDerivative()
302  / (value + 1e-300);
303  } else {
304  double value = pg[i]->evaluate(delta[i]);
305  d2iik(i,j) = pg[i]->getGradient() / (value + 1e-300);
306  value = pg[i]->evaluate(delta[j]);
307  d2iik(i,j) *= pg[i]->getGradient() / (value + 1e-300);
308  }
309  }
310  }
311  for (int i=0; i<NDIM; ++i) {
312  for (int j=0; j<NDIM; ++j) {
313  d2ii(i,j) += mat(ipart,kpart)*d2iik(i,j);
314  }
315  }
316  }
317  for (int i=0; i<NDIM; ++i) {
318  for (int j=0; j<NDIM; ++j) {
319  gradd1(ipart+ifirst,jpart+ifirst)(i)=gradArray1(jpart)(j)*d2ii(j,i);
320  }
321  }
322  } else { // ipart!=jpart
323  // Two derivatives act on different columns, so you get a 2x2 det.
324  Vec g00(0.0),g01(0.0),g10(0.0),g11(0.0);
325  for(int kpart=0; kpart<npart; ++kpart) {
326  // First derivative acts on particle ipart.
327  Vec delta=r1(ipart+ifirst)-r2(kpart+ifirst);
328  cell.pbc(delta);
329  Vec grad, value;
330  for (int i=0; i<NDIM; ++i) {
331  value[i] = pg[i]->evaluate(delta[i]);
332  grad[i] = pg[i]->getGradient()/(value[i] + 1e-300);
333  }
334  for (int i=0; i<NDIM; ++i) {
335  grad *= value[i];
336  }
337  g00 += mat(ipart,kpart)*grad;
338  g10 += mat(jpart,kpart)*grad;
339  // Next derivative acts on particle jpart.
340  delta=r1(jpart+ifirst)-r2(kpart+ifirst);
341  cell.pbc(delta);
342  for (int i=0; i<NDIM; ++i) {
343  value[i] = pg[i]->evaluate(delta[i]);
344  grad[i] = pg[i]->getGradient() / (value[i] + 1e-300);
345  }
346  for (int i=0; i<NDIM; ++i) {
347  grad *= value[i];
348  }
349  g01 += mat(ipart,kpart)*grad;
350  g11 += mat(jpart,kpart)*grad;
351  }
352  for (int i=0; i<NDIM; ++i) {
353  for (int j=0; j<NDIM; ++j) {
354  gradd1(ipart+ifirst,jpart+ifirst)(i)=gradArray1(jpart)(j)
355  *(g00(i)*g11(j)-g10(i)*g01(j));
356  }
357  }
358  }
359  // Then treat d2 terms.
360  // Derivative on row is a sum of derivatives on each column.
361  for(int lpart=0; lpart<npart; ++lpart) {
362  if (ipart==lpart) {
363  // Only one term is non-zero. NEED TO CHECK THESE DERIVATIVES
364  Vec delta=r1(ipart+ifirst)-r2(jpart+ifirst);
365  cell.pbc(delta);
366  Mat d2ii;
367  d2ii = 0.0;
368  for (int i=0; i<NDIM; ++i) {
369  for (int j=0; j<NDIM; ++j) {
370  if (i==j) {
371  double value = pg[i]->evaluate(delta[i]);
372  d2ii(i,j) = pg[i]->getSecondDerivative() / (value + 1e-300);
373  } else {
374  double value = pg[i]->evaluate(delta[i]);
375  d2ii(i,j) = pg[i]->getGradient() / (value + 1e-300);
376  value = pg[j]->evaluate(delta[j]);
377  d2ii(i,j) *= pg[i]->getGradient() / (value + 1e-300);
378  }
379  }
380  }
381  for (int i=0; i<NDIM; ++i) {
382  for (int j=0; j<NDIM; ++j) {
383  d2ii(i,j) *= mat(ipart,jpart);
384  }
385  }
386  for (int i=0; i<NDIM; ++i) {
387  for (int j=0; j<NDIM; ++j) {
388  gradd2(ipart+ifirst,jpart+ifirst)(i)=gradArray2(jpart)(j)*d2ii(j,i);
389  }
390  }
391  } else { // ipart!=lpart
392  // Two derivatives act on different columns, so you get a 2x2 det.
393  Vec g00(0.0),g01(0.0),g10(0.0),g11(0.0);
394  for(int kpart=0; kpart<npart; ++kpart) {
395  // First derivative acts on particle ipart.
396  Vec delta=r1(ipart+ifirst)-r2(kpart+ifirst);
397  cell.pbc(delta);
398  Vec grad, value;
399  for (int i=0; i<NDIM; ++i) {
400  value[i] = pg[i]->evaluate(delta[i]);
401  grad[i] = pg[i]->getGradient() / (value[i] + 1e-300);
402  }
403  for (int i=0; i<NDIM; ++i) {

405  }

408  }

414  value[i] = pg[i]->evaluate(delta[i]);

423  for (int j=0; j<NDIM; ++j) {

437  gradd2(ipart+ifirst,jpart+ifirst)+=gradArray1(ipart);
438  }
439  }
