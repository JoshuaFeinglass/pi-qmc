119  mpi->getCloneComm().Reduce(&norm,&nbuff,1,MPI::DOUBLE,

124  value=vbuff;
125 #endif
126  } else {
127  mpi->getCloneComm().Reduce(&norm,NULL,1,MPI::DOUBLE,MPI::SUM,0);
128  mpi->getCloneComm().Reduce(value.data(),NULL,value.size(),
129  MPI::FLOAT,MPI::SUM,0);
130  }
131  }
132  }
133 #endif
134  // Next add value to accumvalue and accumvalue2.
135  accumvalue += value/norm;
136  if (hasErrorFlag) accumvalue2 += (value*value)/(norm*norm);
137  accumnorm+=1.;

140 }
