208  // First compute the 1D actions.
209  Coulomb1DLinkAction coulomb1D(stau);
210  double u0 = coulomb1D.calculateValueAtOrigin();
211  double u1_0 = coulomb1D.calculateU0(u0, reff);
212  double u1_1 = coulomb1D.calculateU1(reff);
213  double u1_2 = coulomb1D.calculateU2(reff);
214  double u1_3 = coulomb1D.calculateU3(reff);
215  double u1_4 = coulomb1D.calculateU4(reff);
216

245 double CoulombAction::utau(double r, int order) const {
246  double tausave=tau;
