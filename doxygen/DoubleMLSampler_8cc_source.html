116 bool DoubleMLSampler::tryMove(double initialLnTranProb) {
117  double oldDeltaAction = initialLnTranProb;
118  for (int ilevel = nlevel; ilevel >= 0; --ilevel) {
119  if (accRejEst)
120  accRejEst->tryingMove(ilevel);
121 
122  // Make the trial move for this level A2B, and get the acc ratio and transition prob. A2B.
123  factor = defaultFactor;
124  double lnTranProb = mover.makeMove(*this, ilevel);
125  double transA2B = mover.getForwardProb();
126  double deltaAction =
127  (action == 0) ? 0 : action->getActionDifference(*this, ilevel);
128  if (samplingBoth) {
129  activateSection(2);
130  lnTranProb += mover.makeMove(*this, ilevel);
131  transA2B += mover.getForwardProb();
132  deltaAction +=
133  (action == 0) ?
134  0 : action->getActionDifference(*this, ilevel);
135  activateSection(1);
136  }
137  deltaAction += doubleAction->getActionDifference(*this, ilevel);
138  double piRatioBoA = -deltaAction + oldDeltaAction;
139  double acceptProb = exp(lnTranProb + piRatioBoA);
140 
141  // If you want to do DelayedRejection
142  if (delayedRejection && ilevel == nlevel - 1) {
143  if (RandomNumGenerator::getRand() > acceptProb) {
144  double accRatioA2B = acceptProb;
145  for (int i = 0; i < nmoving; ++i) {
146  for (int islice = 0; islice < sectionBeads->getNSlice();
147  ++islice) {
148  (*rejectedBeads1)(i, islice) = (*movingBeads1)(i,
149  islice);
150  (*rejectedBeads2)(i, islice) = (*movingBeads2)(i,
151  islice);
152  }
153  }
154 
155  //Make another trial move for this level A2C, and get the acc ratio.
156  factor = newFactor;
157  lnTranProb = mover.makeMove(*this, ilevel);
158  deltaAction =
159  (action == 0) ?
160  0 : action->getActionDifference(*this, ilevel);
161  if (samplingBoth) {
162  activateSection(2);
163  lnTranProb += mover.makeMove(*this, ilevel);
164  deltaAction +=
165  (action == 0) ?
166  0 :
167  action->getActionDifference(*this, ilevel);
168  activateSection(1);
169  }
170  deltaAction += doubleAction->getActionDifference(*this, ilevel);
171  double piRatioCoA = -deltaAction + oldDeltaAction;
172  double accRatioA2C = exp(lnTranProb + piRatioCoA);
173 
174  //nodal crossing A2C
175  // if (deltaAction > 2e90) return false;
176 
177  //Make the trial move to the rejected state C2B, and get the acc ratio and transition prob. C2B
178  factor = defaultFactor;
179  lnTranProb = mover.makeDelayedMove(*this, ilevel);
180  double transC2B = mover.getForwardProb();
181  if (samplingBoth) {
182  activateSection(2);
183  lnTranProb += mover.makeDelayedMove(*this, ilevel);
184  transC2B += mover.getForwardProb();
185  activateSection(1);
186  }
187  double accRatioC2B = exp(lnTranProb - piRatioCoA + piRatioBoA);
188  accRatioC2B = (accRatioC2B >= 1) ? 1 : accRatioC2B;
189  double accRatioA2B2C = accRatioA2C * exp(transC2B + transA2B)
190  * (1 - accRatioC2B) / (1 - accRatioA2B);
191  deltaAction = deltaAction - transC2B - transA2B
192  - log((1 - accRatioC2B) / (1 - accRatioA2B));
193  if (RandomNumGenerator::getRand() > accRatioA2B2C)
194  return false;
195  }
196  } else {
197  if (RandomNumGenerator::getRand() > acceptProb)
198  return false;
199  }
200  oldDeltaAction = deltaAction;
201 
202  if (accRejEst)
203  accRejEst->moveAccepted(ilevel);
204 
205  }
206

208  action->acceptLastMove();
209 
210  //if (samplingBoth) {
211  // activateSection(2); action->acceptLastMove(); activateSection(1);
212  // }
213  doubleAction->acceptLastMove();
214  // Put moved beads in section beads.
215  for (int islice = 0; islice < sectionBeads->getNSlice(); ++islice) {
216  movingBeads1->copySlice(identityIndex, islice, *sectionBeads1,
