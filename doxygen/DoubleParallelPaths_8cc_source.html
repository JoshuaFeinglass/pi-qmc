163 void DoubleParallelPaths::putBeads(int ifirstSlice, const Beads<NDIM>& inBeads,
164  const Permutation& inPermutation) const {
165  //First place the section beads back in the bead array.
166  bool first = (ifirstSlice-ifirst<nprocSlice+2);
167  Beads<NDIM>& beads(first?beads1:beads2);
168  Permutation& permutation(first?permutation1:permutation2);
169  Permutation& inversePermutation(first
170  ?inversePermutation1:inversePermutation2);
171  int jfirstSlice=(ifirstSlice-ifirst)%(nslice/2);
172  int nsectionSlice=inBeads.getNSlice();
173  for (int isectionSlice=0; isectionSlice<nsectionSlice; ++isectionSlice) {
174  int islice=isectionSlice+jfirstSlice;
175  for (int ipart=0; ipart<npart; ++ipart) {
176  beads(ipart,islice)=inBeads(ipart,isectionSlice);
177  }
178  }
179  // Now permute the following beads.
180  int jlastSlice=nsectionSlice+jfirstSlice;
181  if (!inPermutation.isIdentity()){
182  for (int islice=jlastSlice; islice<nprocSlice+2; ++islice) {
183  // Swap paths.
184  VArray buffer(npart);
185  for (int i=0; i<npart; ++i) buffer(i)=beads(inPermutation[i],islice);
186  for (int i=0; i<npart; ++i) beads(i,islice)=buffer(i);
187  }
188  permutation.prepend(inPermutation);
189  inversePermutation.setToInverse(permutation);
190  }
191 }

238 void DoubleParallelPaths::shift(const int ishift) {
239 #ifdef ENABLE_MPI
240  // Transfer ishift beads between workers.
241  int idest=(iworker+nworker-1)%nworker;
242  int isrc=(iworker+1)%nworker;
243  double *sendbuf1=&(beads1(0,0)[0]);
244  double *recvbuf1=(iworker<nworker-1)?&(buffer1(0,nprocSlice-ishift)[0])
245  :&(buffer2(0,nprocSlice-ishift)[0]);
246  mpi.getWorkerComm().Sendrecv(
247  sendbuf1,(ishift+2)*NDIM*npart,MPI::DOUBLE,idest,1,
248  recvbuf1,(ishift+2)*NDIM*npart,MPI::DOUBLE,isrc,1);
249  double *sendbuf2=&(beads2(0,0)[0]);
250  double *recvbuf2=(iworker<nworker-1)?&(buffer2(0,nprocSlice-ishift)[0])
251  :&(buffer1(0,nprocSlice-ishift)[0]);
252  mpi.getWorkerComm().Sendrecv(
253  sendbuf2,(ishift+2)*NDIM*npart,MPI::DOUBLE,idest,2,
254  recvbuf2,(ishift+2)*NDIM*npart,MPI::DOUBLE,isrc,2);
255  // Permute the ishift beads that we got from the next worker.
256  for (int islice=nprocSlice-ishift; islice<=nprocSlice+1; ++islice) {
257  VArray buffer(npart);
258  for (int i=0; i<npart; ++i) buffer(i)=buffer1(permutation1[i],islice);
259  for (int i=0; i<npart; ++i) buffer1(i,islice)=buffer(i);
260  for (int i=0; i<npart; ++i) buffer(i)=buffer2(permutation2[i],islice);
261  for (int i=0; i<npart; ++i) buffer2(i,islice)=buffer(i);
262  }
263  // Shift the remaining beads that are in this worker.
264  for (int j=0; j<nprocSlice-ishift; ++j) {
265  for (int i=0; i<npart; ++i) {
266  buffer1(i,j)=beads1(i,j+ishift);
267  buffer2(i,j)=beads2(i,j+ishift);
268  }
269  }
270  mpi.getWorkerComm().Barrier();
271  cycleArrays(buffer1.getCoordArray(),beads1.getCoordArray());
272  cycleArrays(buffer2.getCoordArray(),beads2.getCoordArray());
273 #endif
274 }

276 void DoubleParallelPaths::setBuffers() {
277 #ifdef ENABLE_MPI
278  // Get the i=nprocSlice+1 buffer slice from i=1 slice on next worker.
279  int idest=(iworker+nworker-1)%nworker;
280  int isrc=(iworker+1)%nworker;
281  double *sendbuf = &(beads1(0,1)[0]);
282  double *recvbuf = &(buffer1(0,0)[0]);
283  mpi.getWorkerComm().Sendrecv(sendbuf,NDIM*npart,MPI::DOUBLE,idest,1,
284  recvbuf,NDIM*npart,MPI::DOUBLE,isrc,1);
285  // Permute the beads that we got from the next worker.
286  if (iworker<nworker-1) {
287  for (int i=0; i<npart; ++i) {
288  beads1(i,nprocSlice+1)=buffer1(permutation1[i],0);
289  }
290  } else {
291  for (int i=0; i<npart; ++i) {
292  beads2(i,nprocSlice+1)=buffer1(permutation2[i],0);
293  }
294  }
295  sendbuf = &(beads2(0,1)[0]);
296  recvbuf = &(buffer2(0,0)[0]);
297  mpi.getWorkerComm().Sendrecv(sendbuf,NDIM*npart,MPI::DOUBLE,idest,1,
298  recvbuf,NDIM*npart,MPI::DOUBLE,isrc,1);
299  // Permute the beads that we got from the next worker.
300  if (iworker<nworker-1) {
301  for (int i=0; i<npart; ++i) {
302  beads2(i,nprocSlice+1)=buffer2(permutation2[i],0);
303  }
304  } else {
305  for (int i=0; i<npart; ++i) {
306  beads1(i,nprocSlice+1)=buffer2(permutation1[i],0);
307  }
308  }
309  // Get the i=0 buffer slice from i=nprocSlice slice on previous worker.
310  // First permute the beads that we will send to the next worker.
311  if (iworker<nworker-1) {
312  for (int i=0; i<npart; ++i) {
313  buffer1(permutation1[i],0)=beads1(i,nprocSlice);
314  }
315  } else {
316  for (int i=0; i<npart; ++i) {
317  buffer1(permutation2[i],0)=beads2(i,nprocSlice);
318  }
319  }
320  idest=(iworker+1)%nworker;
321  isrc=(iworker+nworker-1)%nworker;
322  sendbuf=&(buffer1(0,0)[0]);
323  recvbuf=&(beads1(0,0)[0]);
324  mpi.getWorkerComm().Sendrecv(sendbuf,NDIM*npart,MPI::DOUBLE,idest,1,
325  recvbuf,NDIM*npart,MPI::DOUBLE,isrc,1);
326  if (iworker<nworker-1) {
327  for (int i=0; i<npart; ++i) {
328  buffer2(permutation2[i],0)=beads2(i,nprocSlice);
329  }
330  } else {
331  for (int i=0; i<npart; ++i) {
332  buffer2(permutation1[i],0)=beads1(i,nprocSlice);
333  }
334  }
335  sendbuf=&(buffer2(0,0)[0]);
336  recvbuf=&(beads2(0,0)[0]);
337  mpi.getWorkerComm().Sendrecv(sendbuf,NDIM*npart,MPI::DOUBLE,idest,1,
338  recvbuf,NDIM*npart,MPI::DOUBLE,isrc,1);
339 #endif
