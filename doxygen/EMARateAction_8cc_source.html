104  Vec re = getMovingPosition(
105  0, islice, nMoving, index, sectionBeads, movingBeads);

134  }

166  }

168  for (int idim = 0; idim < NDIM; ++idim) {
169  oldRadAction += delta[idim] * delta[idim] * inv2Sigma21[idim];
170  }
171  delta=rhRadOld-rhRadPrevOld; cell.pbc(delta);
172  for (int idim = 0; idim < NDIM; ++idim) {
173  oldRadAction += delta[idim] * delta[idim] * inv2Sigma22[idim];
174  }
175

177  Vec deltaPrev = reRadPrevOld - rhRadPrevOld;
178  Vec deltaRad = reRadOld - rhRadOld;
179  Vec deltaDiag = reOld - rhOld;
180  oldDiagAction += coulomb->getValue(deltaDiag, deltaPrev);
181  oldRadAction += coulomb->getValue(deltaRad, deltaPrev);
182  }
183

186  Vec deltaDiag = rePrevOld - rhPrevOld;
187  Vec delta = reOld - rhOld;

189  oldRadAction += coulomb->getValue(delta, deltaRad);
190  }

196  rhRadPrev = rhPrev;

198  rhPrevOld = rhOld;
199  reRadPrevOld = (islice == nSlice / 2) ? rhPrevOld : rePrevOld;

210  }
211  if (log(C) - newRadAction + newDiagAction > 140) {
212  newAction = newRadAction - log(C) - newDiagAction;
213  }
214  }
