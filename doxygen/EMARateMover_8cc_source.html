33  const Beads<NDIM> &movingBeads, const Beads<NDIM> &sectionBeads,
34  int nSlice, const SuperCell& cell) const {

41  Vec mass2 = 0.5 / lambda2;

47  Vec rhRadPrev = rhPrev;

56  Vec rh = movingBeads(1,islice);
57 
58  Vec reRad = re;
59  Vec rhRad = (islice == nSlice/2) ? re : rh;
60

62  for (int idim = 0; idim < NDIM; ++idim) {
63  newDiagAction += delta[idim] * delta[idim] * inv2Sigma21[idim];
64  }

74  for (int idim = 0; idim < NDIM; ++idim) {
75  newRadAction += delta[idim] * delta[idim] * inv2Sigma22[idim];
76  }

79  Vec rhOld = sectionBeads(1,islice);

100

103  rhPrev = rh;
104  reRadPrev = (islice == nSlice / 2) ? rhPrev : rePrev;
105  rhRadPrev = rhPrev;
106  rePrevOld = reOld;
107  rhPrevOld = rhOld;
108  reRadPrevOld = (islice == nSlice / 2) ? rhPrevOld : rePrevOld;
109  rhRadPrevOld = rhPrevOld;
110  }

117  oldAction = oldRadAction - log(C);
118  }

120  newAction = newRadAction - log(C);
121  }
122  }

153 double EMARateMover::calculateRadiatingProbability(
154  const Beads<NDIM> & movingBeads, int nSlice, const SuperCell &cell) {
155  Vec re1 = movingBeads(0, 0);
156  Vec re2 = movingBeads(0, nSlice - 1);
157  Vec rh1 = movingBeads(1, 0);
158  Vec rh2 = movingBeads(1, nSlice - 1);
159  Vec delta1 = re1 - rh1;
160  cell.pbc(delta1);
161  Vec delta2 = re2 - rh2;
162  cell.pbc(delta2);
163  int nStride = nSlice - 1;
164  double inv2sigma21 = 1. / ( (lambda1 + lambda2) * tau * nStride * 2);
165  double t1 = dot(delta1, delta1) * inv2sigma21;
166  double t2 = dot(delta2, delta2) * inv2sigma21;

210 void EMARateMover::sampleRadiating(int nStride, int nSlice,
211  Beads<NDIM> &movingBeads) {
212  for (int iSlice = nStride; iSlice < nSlice; iSlice += 2 * nStride) {
213  blitz::Array<Vec,1> gaussRand(2);
214  Vec mass1 = 0.5 / lambda1;
215  Vec mass2 = 0.5 / lambda2;
216  Vec prev1 = (iSlice - nStride <= nSlice / 2)
217  ? movingBeads(0, iSlice - nStride)
218  : movingBeads(1, nSlice - 1 - (iSlice - nStride));
219  Vec prev2 = (iSlice - nStride < nSlice / 2)
220  ? movingBeads(0, nSlice - 1 - (iSlice - nStride))
221  : movingBeads(1, iSlice - nStride);
222  Vec next1 = (iSlice + nStride <= nSlice / 2)
223  ? movingBeads(0, iSlice + nStride)
224  : movingBeads(1, nSlice -1 - (iSlice + nStride));
225  Vec next2 = (iSlice + nStride < nSlice / 2)
226  ? movingBeads(0, nSlice -1 - (iSlice + nStride))
227  : movingBeads(1, iSlice + nStride);
228 
229  Vec midpoint1, midpoint2;
230  // Should be rewritten to use PBC.
231  if ((iSlice - nStride < nSlice / 2) && (iSlice + nStride > nSlice / 2)) {
232  midpoint1 = (mass1 * prev1 + mass2 * next1) / (mass1 + mass2);
233  midpoint2 = (mass1 * prev2 + mass2 * next2) / (mass1 + mass2);
234  } else {
235  midpoint1 = 0.5 * (prev1 + next1);
236  midpoint2 = 0.5 * (prev2 + next2);
237  }
238  Vec sigma;
239  if (iSlice - nStride < nSlice / 2) {
240  if (iSlice + nStride > nSlice / 2) {
241  sigma = sqrt(0.5 * (lambda1 + lambda2) * tau * nStride);
242  } else {
243  sigma = sqrt(lambda1 * tau * nStride);
244  }
245  }
246  else{
247  sigma = sqrt(lambda2 * tau * nStride);
248  }
249  makeGaussianRandomNumbers(gaussRand);

251  Vec delta2 = gaussRand(1) * sigma;
252  if (iSlice < nSlice / 2){
253  movingBeads(0, iSlice) = midpoint1 + delta1;
254  movingBeads(0, nSlice - 1 - iSlice) = midpoint2 + delta2;
255  } else {
256  if (iSlice == nSlice / 2){
257  movingBeads(0, iSlice) = midpoint1 + delta1;
258  movingBeads(1, iSlice) = midpoint2 + delta2;
259  } else {

261  movingBeads(1, nSlice - 1 - iSlice) = midpoint1 + delta1;

263  }
264  }

267 void EMARateMover::sampleDiagonal(int nStride, int nSlice,
268  Beads<NDIM> &movingBeads, const SuperCell & cell) {
269  for (int iSlice = nStride; iSlice < nSlice; iSlice += 2 * nStride) {
270  blitz::Array<Vec,1> gaussRand(2);
271  makeGaussianRandomNumbers(gaussRand);
272  for(int iMoving = 0;iMoving < 2;++iMoving){
273  double lambda = (iMoving == 0) ? lambda1 : lambda2;
274  double sigma = sqrt(lambda * tau * nStride);
275  // Calculate the new position.
276  Vec midpoint = movingBeads.delta(iMoving, iSlice + nStride, -2 * nStride);
277  cell.pbc(midpoint) *= 0.5;
278  midpoint += movingBeads(iMoving, iSlice - nStride);
279  Vec delta = gaussRand(iMoving);
280  delta *= sigma;
281  movingBeads(iMoving, iSlice) = midpoint + delta;
282  cell.pbc(movingBeads(iMoving, iSlice));
283  }
284  }
