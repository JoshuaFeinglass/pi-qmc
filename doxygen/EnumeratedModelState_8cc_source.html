35 bool EnumeratedModelState::read(const std::string &line) {
36  std::cout << "Checking for model state..." << std::endl;
37  int i = line.find("state");
38  bool didRead = false;
39  if (i != -1) {
40  i += 6;
41  int j = line.find("of");
42  modelState = atoi(line.substr(i,j-i-1).c_str());
43  //modelCount = atoi(line.substr(j+3).c_str());
44  // Hack to handle old and new indexing of model states.
45  // New convention is to start from 1 and end line with "."
46  if (line[line.size()-1]=='.') --modelState;
47  std::cout << "Model state " << modelState << "." << std::endl;
48  didRead = true;
49  }
50  return didRead;
51 }
