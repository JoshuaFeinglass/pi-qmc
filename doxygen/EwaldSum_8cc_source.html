106  // Set up the exponential tables
107 #if NDIM==3 || NDIM==2
108  const Complex I(0,1);
109  for (int ipart=0; ipart<npart; ++ipart) {
110  eikx(0,ipart)=1.0;

112 #if NDIM==3
113  eikz(0,ipart)=1.0;
114 #endif
115  }
116  if (ikmax[0]>0) {

119  }
120  }
121  if (ikmax[1]>0) {
122  for (int ipart=0; ipart<npart; ++ipart) {
123  eiky(1,ipart)=exp(I*deltak[1]*r(ipart)[1]);

125  }
126  }
127 #if NDIM==3
128  if (ikmax[2]>0) {
129  for (int ipart=0; ipart<npart; ++ipart) {
130  eikz(1,ipart)=exp(I*deltak[2]*r(ipart)[2]);
131  eikz(-1,ipart)=conj(eikz(1,ipart));
132  }
133  }
134 #endif
135  for (int kx=2; kx<=ikmax[0]; ++kx) {
136  for (int ipart=0; ipart<npart; ++ipart) {
137  eikx(kx,ipart)=eikx(kx-1,ipart)*eikx(1,ipart);
138  }
139  }
140  for (int ky=2; ky<=ikmax[1]; ++ky) {
141  for (int ipart=0; ipart<npart; ++ipart) {
142  eiky(ky, ipart)=eiky(ky-1, ipart)*eiky(1,ipart);
143  eiky(-ky, ipart)=eiky(-ky+1, ipart)*eiky(-1,ipart);
144  }
145  }
146 #if NDIM==3

151  }
152  }
153 #endif

155 #ifdef _OPENMP
