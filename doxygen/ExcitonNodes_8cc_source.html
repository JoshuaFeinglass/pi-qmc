81 NodeModel::DetWithFlag
82 ExcitonNodes::evaluate(const VArray &r1, const VArray &r2,
83  const int islice, bool scaleMagnitude) {
84  DetWithFlag result; result.err=false;
85  do { // Loop if scale is wrong to avoid overflow/underflow.
86  Matrix& mat(*matrix[islice]);
87  mat=0;
88  for(int jpart=0; jpart<npart; ++jpart) {
89  for(int ipart=0; ipart<npart; ++ipart) {
90  Vec delta(r1(jpart+ifirst)-r1(ipart+jfirst));
91  cell.pbc(delta);
92  double ear=exp(-alpha*sqrt(dot(delta,delta)));
93  mat(ipart,jpart)=scale*ear+1e-100;
94  uarray(ipart,jpart)=-log(fabs(mat(ipart,jpart))+1e-100);
95  }
96  }
97  // Find dominant contribution to determinant (distroys uarray).
98  hungarian->solve(uarray.data());
99  for (int ipart=0; ipart<npart; ++ipart) {
100  kindex(islice,ipart) = (*hungarian)[ipart];
101  }
102  // Note: u(ipart,jpart=kindex(islice,ipart)) makes maximum contribution
103  // or lowest total action.
104  // Next calculate determinant and inverse.
105  int info=0;//LU decomposition
106  DGETRF_F77(&npart,&npart,mat.data(),&npart,ipiv.data(),&info);
107  if (info!=0) {
108  result.err=true;
109  std::cout << "BAD RETURN FROM ZGETRF!!!!" << std::endl;
110  nerror++;
111  if (nerror>1000) {
112  std::cout << "too many errors!!!!" << std::endl;
113  std::exit(-1);
114  }
115  }
116  double det = 1;
117  for (int i=0; i<npart; ++i) {
118  det*= mat(i,i);
119  det *= (i+1==ipiv(i))?1:-1;
120  }
121  DGETRI_F77(&npart,mat.data(),&npart,ipiv.data(),work.data(),&lwork,&info);
122  if (info!=0) {
123  result.err = true;
124  std::cout << "BAD RETURN FROM ZGETRI!!!!" << std::endl;
125  nerror++;
126  if (nerror>1000) {
127  std::cout << "too many errors!!!!" << std::endl;
128  std::exit(-1);
129  }
130  }
131  // watch for overflow or underflow.
132  if (scaleMagnitude && (!result.err
133  && (fabs(det)<1e-50 || fabs(det)>1e50) )) {
134  if (fabs(det)<1e-250) {
135  scale *= 2;
136  } else {
137  scale *= pow(fabs(det),-1./npart);
138  }
139  std::cout << "Slater determinant rescaled: scale = "
140  << scale << std::endl;
141  }
142  result.det = det;
143  }
144  while (scaleMagnitude && (result.err ||
145  (fabs(result.det)<1e-50 || fabs(result.det)>1e50)));
146  return result;
147 }

154 void ExcitonNodes::evaluateDistance(const VArray& r1, const VArray& r2,
155  const int islice, Array& d1, Array& d2) {
156  Matrix& mat(*matrix[islice]);
157  // Calculate log gradients to estimate distance.
158  d1=200; d2=200; // Initialize distances to a very large value.
159  for (int jpart=0; jpart<npart; ++jpart) {
160  Vec logGrad=0.0, fgrad=0.0;
161  for (int ipart=0; ipart<npart; ++ipart) {
162  Vec delta(r1(jpart+ifirst)-r1(ipart+jfirst));
163  cell.pbc(delta);
164  double d=sqrt(dot(delta,delta));
165  Vec grad = -alpha*delta/(d+1e-200);
166  if (ipart==kindex(islice,jpart)) fgrad=grad;
167  grad *= exp(-alpha*d);
168  logGrad += mat(jpart,ipart)*scale*grad;
169  }
170  gradArray1(jpart)=logGrad-fgrad;
171  d1(jpart+ifirst)
172  =sqrt(2*mass/((dot(gradArray1(jpart),gradArray1(jpart))+1e-15)*tau));
173  }
174  for (int ipart=0; ipart<npart; ++ipart) {
175  Vec logGrad=0.0, fgrad=0.0;
176  for (int jpart=0; jpart<npart; ++jpart) {
177  Vec delta(r1(ipart+jfirst)-r1(jpart+ifirst));
178  cell.pbc(delta);
179  double d=sqrt(dot(delta,delta));
180  Vec grad = -alpha*delta/(d+1e-200);
181  if (ipart==kindex(islice,jpart)) fgrad=grad;
182  grad *= exp(-alpha*d);
183  logGrad += mat(ipart,jpart)*scale*grad;
184  }
185  gradArray1(ipart)=logGrad-fgrad;
186  d1(ipart+jfirst)
187  =sqrt(2*mass/((dot(gradArray1(ipart),gradArray1(ipart))+1e-15)*tau));
188  }
189 }
