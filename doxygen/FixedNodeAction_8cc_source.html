254 void FixedNodeAction::getBeadAction(const Paths &paths, int ipart, int islice,
255  double &u, double &utau, double &ulambda, Vec &fm, Vec &fp) const {
256  int totNSlice=paths.getNSlice();
257  fm=0; fp=0; u=utau=ulambda=0;
258  // Attribute u and utau to first particle.
259  // We only calculate determinants when iPart==0, then store
260  // the forces in the force array.
261  if (withNodalAction && ipart==0) {
262  // Calculate the action and the gradient of the action.
263  // Calculate d_i-1
264  int jslice=(islice+totNSlice/2)%totNSlice;
265  for (int i=0; i<npart; ++i) r1(i)=paths(i,islice,-1);
266  for (int i=0; i<npart; ++i) r2(i)=paths(i,jslice,-1);
267  NodeModel::DetWithFlag
268  detm = nodeModel->evaluate(r1, r2, 0, false);
269  nodeModel->evaluateDistance(r1,r2,0,dim1,dim2);
270  if (useDistDerivative) {
271  nodeModel->evaluateDotDistance(r1,r2,0,dotdim1,dotdim2);
272  } else {
273  dotdim1=0.; dotdim2=0.;
274  }
275  // Calculate the action and the gradient of the action.
276  for (int i=0; i<npart; ++i) r1(i)=paths(i,islice);
277  for (int i=0; i<npart; ++i) r2(i)=paths(i,jslice);
278  NodeModel::DetWithFlag
279  det = nodeModel->evaluate(r1, r2, 0, false);
280  nodeModel->evaluateDistance(r1,r2,0,di1,di2);
281  if (useDistDerivative) {
282  nodeModel->evaluateDotDistance(r1,r2,0,dotdi1,dotdi2);
283  } else {
284  dotdi1=0.; dotdi2=0.;
285  }
286  // For now, forces are zero, so we cannot use the virial estimator.
287  force=0.0;
288  // Calculate u and utau.
289  if (useManyBodyDistance) {
290  double d12=0., d1m2=0., dotd11=0., dotd1m1=0.;
291  for (int i=0; i<npart; ++i) {
292  d12 += 1./(di1(i)*di1(i));
293  d1m2 += 1./(dim1(i)*dim1(i));
294  dotd11 += dotdi1(i)/(di1(i)*di1(i)*di1(i));
295  dotd1m1 += dotdim1(i)/(dim1(i)*dim1(i)*dim1(i));
296  }
297  d12 = 1./sqrt(d12);
298  d1m2 = 1./sqrt(d1m2);
299  dotd11 *= d12*d12*d12;
300  dotd1m1 *= d1m2*d1m2*d1m2;
301  double xim1 = d12*d1m2;
302  double exim1 = exp(-xim1);
303  u -= log( (1-exim1) );
304  utau -= (dotd11*d1m2 + d12*dotd1m1)*exim1/(1-exim1);
305  } else {
306  for (int jpart=0; jpart<npart; ++jpart) {
307  double xim1=dim1(jpart)*di1(jpart);
308  double dotxim1=dotdim1(jpart)*di1(jpart)+dim1(jpart)*dotdi1(jpart);
309  // Calculate the nodal action.
310  u += -log(1-exp(-xim1));
311  if (det.err || detm.err || det.det*detm.det<0) u += 1e200;
312  utau += -dotxim1*exp(-xim1)/(1-exp(-xim1));
313  }
314  }
315  } else {
316  // Just check for node crossing.
317  int jslice=(islice+totNSlice/2)%totNSlice;
318  for (int i=0; i<npart; ++i) r1(i)=paths(i,islice,-1);
319  for (int i=0; i<npart; ++i) r2(i)=paths(i,jslice,-1);
320  NodeModel::DetWithFlag
321  detm = nodeModel->evaluate(r1, r2, 0, false);
322  for (int i=0; i<npart; ++i) r1(i)=paths(i,islice);
323  for (int i=0; i<npart; ++i) r2(i)=paths(i,jslice);
324  NodeModel::DetWithFlag
325  det = nodeModel->evaluate(r1, r2, 0, false);
326  if (det.err || detm.err ||det.det*detm.det < 0) u = 1e200;
327  }
328  fm=force(ipart);
329 }
