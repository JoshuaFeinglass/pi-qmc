121 double FixedPhaseAction::action(const double phi0, const double phi1,
122  const VArray& r1, const VArray& r2,
123  const VArray& gradPhi01, const VArray& gradPhi11,
124  const VArray& gradPhi02, const VArray& gradPhi12,
125  const VArray& a01, const VArray& a11,
126  const VArray& a02, const VArray& a12,
127  const double teff, const bool reject) const {
128  // Calculate change in phase; optionally reject if greater than one radian.
129  double deltaPhi=phi1-phi0;
130  //deltaPhi -= (((int)floor(deltaPhi/PI+1.0))/2)*(2.0*PI);
131  deltaPhi -= floor(0.5*(deltaPhi/PI+1.0))*2.0*PI;
132  if (reject && fabs(deltaPhi)>1.5) return 1e200;
133  // Calculate the distance propagated.
134  double d=1e-20;
135  for (int i=0; i<npart; ++i) {
136  d+=dot(r1(i),r1(i))+dot(r2(i),r2(i));
137  }
138  d = sqrt(d);
139  double dinv=1.0/d;
140  // Calculate the partial derivatives parallel to the displacement.
141  double parPhi0=0,parPhi1=0,parA0=0,parA1=0;
142  for (int i=0; i<npart; ++i) {
143  parPhi0 += dot(gradPhi01(i),r1(i))+dot(gradPhi02(i),r2(i));
144  parPhi1 += dot(gradPhi11(i),r1(i))+dot(gradPhi12(i),r2(i));
145  parA0 += dot(a01(i),r1(i))+dot(a02(i),r2(i));
146  parA1 += dot(a11(i),r1(i))+dot(a12(i),r2(i));
147  }
148  parPhi0*=dinv; parPhi1*=dinv; parA0*=dinv; parA1*=dinv;
149  // Calculate the action.
150  double action = 1.2*deltaPhi*deltaPhi*dinv*dinv;
151  action += (2*parPhi0*parPhi0+2*parPhi1*parPhi1-parPhi0*parPhi1)/15.0;
152  action -= 0.2*deltaPhi*(parPhi1+parPhi0)*dinv;
153  action -= (parA1+parA0)*deltaPhi*dinv;
154  action += (parA1-parA0)*(parPhi1-parPhi0)/6.0;
155  action += (parA1*parA1+parA0*parA0+parA1*parA0)/3.0;
156  if(action<0) std::cout << "ERROR: FixedPhaseAction already negative" << std::endl;
157  for (int i=0; i<npart; ++i) {
158  //action += 0.5*dot(gradPhi11(i)-a11(i),gradPhi11(i)-a11(i));
159  //action += 0.5*dot(gradPhi01(i)-a01(i),gradPhi01(i)-a01(i));
160  //action += 0.5*dot(gradPhi12(i)+a12(i),gradPhi12(i)+a12(i));
161  //action += 0.5*dot(gradPhi02(i)+a02(i),gradPhi02(i)+a02(i));
162  Vec temp = gradPhi11(i)-a11(i)-gradPhi01(i)+a01(i)
163  -r1(i)*dinv*(parPhi1-parA1-parPhi0+parA0);
164  action += dot(temp,temp)/3.0;
165  Vec temp1 = (gradPhi11(i)-a11(i))-r1(i)*dinv*(parPhi1-parA1);
166  Vec temp0 = (gradPhi01(i)-a01(i))-r1(i)*dinv*(parPhi0-parA0);
167  action += dot(temp1,temp0);
168  temp = gradPhi12(i)-a12(i)-gradPhi02(i)+a02(i)
169  -r2(i)*dinv*(parPhi1-parA1-parPhi0+parA0);
170  action += dot(temp,temp)/3.0;
171  temp1 = (gradPhi12(i)-a12(i))-r2(i)*dinv*(parPhi1-parA1);
172  temp0 = (gradPhi02(i)-a02(i))-r2(i)*dinv*(parPhi0-parA0);
173  action += dot(temp1,temp0);
174  }
175  action *= teff/(2*mass);
176  if(action<0) std::cout << "ERROR: FixedPhaseAction negative" << std::endl;
177  return action;
178 }

180 void FixedPhaseAction::getBeadAction(const Paths &paths, int ipart, int islice,
181  double &u, double &utau, double &ulambda, Vec &fm, Vec &fp) const {
182  int totNSlice=paths.getNSlice();
183  fm=0; fp=0; u=utau=ulambda=0;
184  // Attribute u and utau to first particle.
185  // We only calculate determinants when iPart==0, then store
186  // the forces in the force array.
187  if (ipart==0) {
188  int jslice=(islice+totNSlice/2)%totNSlice;
189  for (int i=0; i<npart; ++i) r1(i)=paths(i,islice);
190  for (int i=0; i<npart; ++i) r2(i)=paths(i,jslice);
191  phaseModel->evaluate(r1,r2,1);
192  newPhi(1) = phaseModel->getPhi();
193  newGradPhi1[1] = phaseModel->getGradPhi(1);
194  newGradPhi2[1] = phaseModel->getGradPhi(2);
195  newVecPot1[1] = phaseModel->getVecPot(1);
196  newVecPot2[1] = phaseModel->getVecPot(2);
197  for (int i=0; i<npart; ++i) r1(i)=paths(i,islice,-1);
198  for (int i=0; i<npart; ++i) r2(i)=paths(i,jslice,-1);
199  phaseModel->evaluate(r1,r2,0);
200  newPhi(0) = phaseModel->getPhi();
201  newGradPhi1[0] = phaseModel->getGradPhi(1);
202  newGradPhi2[0] = phaseModel->getGradPhi(2);
203  newVecPot1[0] = phaseModel->getVecPot(1);
204  newVecPot2[0] = phaseModel->getVecPot(2);
205  utau = 0.5*action(newPhi(0),newPhi(1),r1,r2,
206  newGradPhi1[0],newGradPhi1[1], newGradPhi2[0],newGradPhi2[1],
207  newVecPot1[0],newVecPot1[1], newVecPot2[0],newVecPot2[1], tau)/tau;
208  }
