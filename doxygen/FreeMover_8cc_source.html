61 double FreeMover::makeMove(MultiLevelSampler& sampler, const int level) {
62  const Beads<NDIM>& sectionBeads = sampler.getSectionBeads();
63  Beads<NDIM>& movingBeads = sampler.getMovingBeads();
64  const SuperCell& cell = sampler.getSuperCell();
65  const int nStride = (int) pow(2, level);
66  const int nSlice = sectionBeads.getNSlice();
67  double factor = sampler.getFactor();
68  const blitz::Array<int, 1>& index = sampler.getMovingIndex();
69  const int nMoving = index.size();
70  blitz::Array<Vec, 1> gaussRand(nMoving);
71  gaussRand = 0.0;
72  double toldOverTnew = 0;
73  forwardProb = 0;
74  for (int islice = nStride; islice < nSlice - nStride;
75  islice += 2 * nStride) {
76  RandomNumGenerator::makeGaussRand(gaussRand);
77  for (int iMoving = 0; iMoving < nMoving; ++iMoving) {
78  const int i = index(iMoving);
79  const int ispec = specIndex(i);
80  double sigma = factor * sqrt(lambda(i) * tau * nStride);
81  double inv2Sigma2 = 0.5 / (sigma * sigma);
82  // Calculate the new position.
83  Vec midpoint = movingBeads.delta(iMoving, islice + nStride,
84  -2 * nStride);
85  cell.pbc(midpoint) *= 0.5;
86  midpoint += movingBeads(iMoving, islice - nStride);
87  cell.pbc(midpoint);
88  Vec delta = gaussRand(iMoving);
89  delta *= sigma;
90  cell.pbc(delta);
91  (movingBeads(iMoving, islice) = midpoint) += delta;
92  cell.pbc(movingBeads(iMoving, islice));
93  // Calculate 1/transition probability for move
94  double temp = 1.0;
95  for (int idim = 0; idim < NDIM; ++idim) {
96  if (pg(level, ispec, idim)) {
97  temp *= pg(level, ispec, idim)->evaluate(delta[idim]);
98  } else {
99  double tmp = delta[idim] * delta[idim] * inv2Sigma2;
100  forwardProb += tmp;
101  toldOverTnew += tmp;
102  }
103  }
104  temp = 1.0 / temp;
105  forwardProb += log(temp);
106 
107  // Calculate and add reverse transition probability.
108  midpoint = sectionBeads.delta(i, islice + nStride, -2 * nStride);
109  cell.pbc(midpoint) *= 0.5;
110  midpoint += sectionBeads(i, islice - nStride);
111  cell.pbc(midpoint);
112  delta = sectionBeads(i, islice);
113  delta -= midpoint;
114  cell.pbc(delta);
115  for (int idim = 0; idim < NDIM; ++idim) {
116  if (pg(level, ispec, idim)) {
117  temp *= pg(level, ispec, idim)->evaluate(delta[idim]);
118  } else {
119  toldOverTnew -= delta[idim] * delta[idim] * inv2Sigma2;
120  }
121  }
122  toldOverTnew += log(temp);
123  }
124  }
125  return toldOverTnew; //Return the log of the probability.
126 }
