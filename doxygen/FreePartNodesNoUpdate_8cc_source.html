108  const Beads<NDIM>& sectionBeads2=sampler.getSectionBeads(2);

110  const Beads<NDIM>& movingBeads2=sampler.getMovingBeads(2);
111  const IArray& index1=sampler.getMovingIndex(1);
112  const IArray& index2=sampler.getMovingIndex(2);
113  if (index1(0)<ifirst||index1(0)>=ifirst+npart) {notMySpecies=true; return 0;}
114  else notMySpecies=false;
115  int nSlice=sectionBeads1.getNSlice();
116  const int nMoving=index1.size();
117  const int nStride = 1 << (level + 1);
118  // First check for node crossing.
119  for (int islice=nStride/2; islice<nSlice; islice+=nStride) {
120  for (int i=0; i<npart; ++i) r1(i)=sectionBeads1(i+ifirst,islice);
121  for (int i=0; i<nMoving; ++i) r1(index1(i)-ifirst)=movingBeads1(i,islice);
122  for (int i=0; i<npart; ++i) r2(i)=sectionBeads2(i+ifirst,islice);
123  if (sampler.isSamplingBoth()) for (int i=0; i<nMoving; ++i)
124  r2(index2(i)-ifirst)=movingBeads2(i,islice);
125  newDet(islice)=evaluate(r1,r2,islice);
126  if (newDet(islice)*det(0)<=0) return deltaAction=2e100;
127  }
128  // Calculate the nodal action if level=0;
129  if (level==0) {
130  const double twomovert=2*mass/tau;
131  double d0=0,newd0=0,d1=0,newd1=0;
132  for (int islice=0;islice<nSlice; ++islice) {
133  newd1=d1=dist(islice);
134  if (islice>0) {
135  for (int i=0; i<npart; ++i) r1(i)=sectionBeads1(i+ifirst,islice);
136  for (int i=0; i<nMoving; ++i)
137  r1(index1(i)-ifirst)=movingBeads1(i,islice);
138  for (int i=0; i<npart; ++i) r2(i)=sectionBeads2(i+ifirst,islice);
139  if (sampler.isSamplingBoth()) for (int i=0; i<nMoving; ++i)
140  r2(index2(i)-ifirst)=movingBeads2(i,islice);
141  newd1=newDist(islice)=evaluateDistance(islice);
142  deltaAction+=log( (1-exp(-twomovert*d1*d0))
143  /(1-exp(-twomovert*newd1*newd0)) );
144 // deltaAction+=-log(1-exp(-twomovert*newd1*newd0))
145 // +log(1-exp(-twomovert*d1*d0));

147  }

149  }

191  // Attribute u and utau to first particle.
192  // We only calculate determinants when iPart==ifirst, then store
193  // the forces in the force array.
194  if (iPart==ifirst) {
195  // Calculate the action and the gradient of the action.
196  // Calculate d_i-1

198  for (int i=0; i<npart; ++i) r1(i)=paths(i+ifirst,islice,-1);

200  evaluate(r1, r2, 0);

204  for (int i=0; i<npart; ++i) r2(i)=paths(i+ifirst,jslice,+1);
205  evaluate(r1, r2, 0);
206  double dip1=evaluateDistance(0);
207  // Calculate the action and the gradient of the action.
208  for (int i=0; i<npart; ++i) r1(i)=paths(i+ifirst,islice);
209  for (int i=0; i<npart; ++i) r2(i)=paths(i+ifirst,jslice);
210  evaluate(r1, r2, 0);
211  double di=evaluateDistance(0);
212  // Now calculate vector to node to get derivatives of action.
213  const Matrix& mat(*matrix[0]);
214  gradArray=0.0;
215  gradMatrix=0.0;
216  for (int jpart=0; jpart<npart; ++jpart) {
217  for (int ipart=0; ipart<npart; ++ipart) {
218  grad2Matrix(ipart,jpart)=0.;
219  // First calculate first derivative (loggrad) terms.
220  Vec delta=r1(jpart)-r2(ipart);
221  cell.pbc(delta);
222  Vec grad, value;
223  for (int i=0; i<NDIM; ++i) {
224  value[i] = pg.evaluate(delta[i]);
225  grad[i] = pg.getGradient() / (value[i] + 1e-300);
226  }
227  for (int i=0; i<NDIM; ++i) {
228  grad *= value[i];
229  }
230  gradArray(jpart)+=mat(jpart,ipart)*grad;
231  // Calculate more complicated second derivative terms.
232  for (int kpart=0; kpart<npart; ++kpart) {
233  if (kpart==jpart) {
234  Mat grad2;
235  for (int i=0; i<NDIM; ++i) {
236  for (int j=0; j<NDIM; ++j) {
237  if (i==j) {
238  double value = pg.evaluate(delta[i]);
239  grad2(i,i) = pg.getSecondDerivative()
240  / (value + 1e-300);
241  } else {
242  double value = pg.evaluate(delta[i]);
243  grad2(i,j) = pg.getGradient() / (value + 1e-300);
244  value = pg.evaluate(delta[j]);
245  grad2(i,j) *= pg.getGradient() / (value + 1e-300);
246  }
247  }
248  }
249  for (int ii=0; ii<NDIM; ++ii) {
250  for (int i=0; i<NDIM; ++i) {
251  for (int j=0; j<NDIM; ++j) {

253  }
254  }
255  }

257  for (int j=0; j<NDIM; ++j) {
258  grad2Matrix(jpart,jpart)(i,j) += mat(jpart,ipart)*grad2(i,j);
259  }
260  }
261  } else {
262  gradMatrix(jpart,kpart)+=mat(kpart,ipart)*grad;
263  }
264  }
265  }

268  for (int jpart=0; jpart<npart; ++jpart) {
269  for (int kpart=0; kpart<npart; ++kpart) {
270  if (kpart!=jpart) {

276  }
277  }
278  }
279  }
280  }

286  }
287  }
288  }
289  }
290  // Calculate terms with gradients on other beads.
291  for (int jpart=0; jpart<npart; ++jpart) {
292  gradArray=0.0;
293  for (int kpart=0; kpart<npart; ++kpart) {
294  Vec delta=r2(jpart)-r1(kpart);
295  cell.pbc(delta);
296  Vec grad, value;
297  for (int i=0; i<NDIM; ++i) {

302  grad *= value[i];
303  }
304  for (int ipart=0; ipart<npart; ++ipart) {
305  gradArray(ipart)+=mat(kpart,ipart)*grad;
306  }
307  }

309  for (int kpart=0; kpart<npart; ++kpart) {
310  mat2(kpart,jpart)=mat(kpart,jpart)*gradArray(jpart)
311  /dot(gradArray(jpart),gradArray(jpart));
312  }
313  for (int ipart=0; ipart<npart; ++ipart) {
314  if (ipart!=jpart) {
315  for (int kpart=0; kpart<npart; ++kpart) {
316  mat2(kpart,ipart)=mat(kpart,ipart);
317  for (int i=0; i<NDIM; ++i) {
318  mat2(kpart,ipart)[i]-=mat2(kpart,jpart)[i]*gradArray(ipart)[i];
319  }
320  }
321  }
322  }

324  for (int ipart=0; ipart<npart; ++ipart) {
325  Vec temp=0.0;
326  for (int kpart=0; kpart<npart; ++kpart) {
327  Vec delta=r1(ipart)-r2(kpart);
328  cell.pbc(delta);
329  if (jpart==kpart) {

332  for (int j=0; j<NDIM; ++j) {
333  if (i==j) {
334  double value = pg.evaluate(delta[i]);
335  grad2(i,j) = -pg.getSecondDerivative() / (value + 1e-300);
336  } else {
337  double value = pg.evaluate(delta[i]);
338  grad2(i,j) = -pg.getGradient() / (value + 1e-300);
339  value = pg.evaluate(delta[j]);
340  grad2(i,j) *= pg.getGradient() / (value + 1e-300);
341  }
342  }
343  }
344  for (int ii=0; ii<NDIM; ++ii) {
345  for (int i=0; i<NDIM; ++i) {
346  for (int j=0; j<NDIM; ++j) {

348  }
349  }

352  for (int j=0; j<NDIM; ++j) {
353  temp(i) +=mat(ipart,jpart)*grad2(i,j)*gradArray(jpart)(j);
354  }
355  }
356  } else {

358  for (int i=0; i<NDIM; ++i) {
359  value[i] = pg.evaluate(delta[i]);
360  grad[i] = pg.getGradient() / (value[i] + 1e-300);
361  }
362  for (int i=0; i<NDIM; ++i) {
363  grad *= value[i];
364  }
365  temp+=grad*dot(mat2(ipart,kpart),gradArray(jpart));
366  }
367  force(ipart)-=di*di*temp;
368  }
369  }
370  }
371  // Now calulate factor on force.
372  double xip1=2*mass*dip1*di/tau;
373  double xim1=2*mass*dim1*di/tau;

377  // Calculate the nodal action.
378  u = -0.5*log(1-exp(-xim1));
