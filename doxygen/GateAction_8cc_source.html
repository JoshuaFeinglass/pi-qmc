33 /*
34 This is a potential mimicking the squre gate potential in experiments.
35 The potential is GVolt * ((tanh(sx(x + xwidth/2)) - tanh(sx(x - xwidth/2))))
36  * ((tanh(sy(y + ywidth/2)) - tanh(sy(y - ywidth/2))))
37 GVolt is the potential of the top gate in energy unit.
38 s is a parameter controling the flatness of the top of the potential.
39 offset is a parameter representing the offset w.r.t. x-axis and y-axis.

56 double GateAction::getActionDifference(const SectionSamplerInterface& sampler, const int level) {
57  const Beads<NDIM>& sectionBeads=sampler.getSectionBeads();
58  const Beads<NDIM>& movingBeads=sampler.getMovingBeads();
59  const SuperCell& cell=sampler.getSuperCell();
60  const int nStride = 1 << level;
61  const int nSlice=sectionBeads.getNSlice();
62  const IArray& index=sampler.getMovingIndex();
63  const int nMoving=index.size();
64  double deltaAction=0;
65  double ktstride = tau*nStride;
66  double xspread = xwidth / 2;
67  double yspread = ywidth / 2;
68  for (int islice=nStride; islice<nSlice-nStride; islice+=nStride) {
69  for (int iMoving=0; iMoving<nMoving; ++iMoving) {
70  const int i=index(iMoving);
71  if (i<ifirst || i>+ifirst+npart) continue;
72  // Add action for moving beads.
73  Vec delta=movingBeads(iMoving, islice);
74  cell.pbc(delta);
75  double x=0, y=0;
76  x = delta[0];
77  y = delta[1];
78  deltaAction += GVolt * (tanh(sx * (x - xoffset + xspread)) -
79  tanh(sx * (x - xoffset - xspread))) *
80  (tanh(sy * (y - yoffset + yspread)) -
81  tanh(sy * (y - yoffset - yspread))) * ktstride
82  / normalConst;
83  // Subtract action for old beads.
84  delta = sectionBeads(i, islice);
85  cell.pbc(delta);
86  x = delta[0];
87  y = delta[1];
88  deltaAction -= GVolt * (tanh(sx * (x - xoffset + xspread)) -
89  tanh(sx * (x - xoffset - xspread))) *
90  (tanh(sy * (y - yoffset + yspread)) -
91  tanh(sy * (y - yoffset - yspread))) * ktstride
92  / normalConst;
93  }
94  }
95  return deltaAction;
96 }

98 double GateAction::getActionDifference(const Paths& paths,
99  const VArray& displacement,
100  const int nMoving, const IArray& movingIndex,
101  int iFirstSlice, int iLastSlice) {
102  double deltaAction = 0.;
103  const SuperCell& cell = paths.getSuperCell();
104  double xspread = xwidth / 2;
105  double yspread = ywidth / 2;
106  for (int i=0; i<nMoving; ++i) {
107  int ipart = movingIndex(i);
108  if (ipart<ifirst || ipart>ifirst+npart) break;
109  for (int islice = iFirstSlice; islice <= iLastSlice; ++islice) {
110  Vec r = paths(ipart, islice);
111  double x = r[0];
112  double y = r[1];
113  deltaAction -= tau * GVolt * (tanh(sx * (x - xoffset + xspread)) -
114  tanh(sx * (x - xoffset - xspread))) *
115  (tanh(sy * (y - yoffset + yspread)) -
116  tanh(sy * (y - yoffset - yspread)))
117  / normalConst;
118  r += displacement(i);
119  cell.pbc(r);
120  x = r[0];
121  y = r[1];
122  deltaAction += tau * GVolt * (tanh(sx * (x - xoffset + xspread)) -
123  tanh(sx * (x - xoffset - xspread))) *
124  (tanh(sy * (y - yoffset + yspread)) -
125  tanh(sy * (y - yoffset - yspread)))
126  / normalConst;
127  }
128  }
129  return deltaAction;
130 }

136 void GateAction::getBeadAction(const Paths& paths, int ipart, int islice, double& u, double& utau, double& ulambda, Vec &fm, Vec &fp) const {
137  Vec delta = paths(ipart,islice);
138  fm = 0; fp = 0; ulambda = 0;
139  if (ipart < ifirst || ipart >= ifirst + npart) return;
140  double x = delta[0];
141  double y = delta[1];
142  double xspread = xwidth / 2;
143  double yspread = ywidth / 2;
144  utau = GVolt * (tanh(sx * (x - xoffset + xspread)) -
145  tanh(sx * (x - xoffset - xspread))) *
146  (tanh(sy * (y - yoffset + yspread)) -
147  tanh(sy * (y - yoffset - yspread))) ;
148  u = utau * tau / normalConst;
149 }
