120 void H5ReportBuilder::writeStep(EstimatorManager& manager) {
121  dset=dataset.begin();
122  for (EstimatorManager::EstimatorIter est=manager.estimator.begin();
123  est!=manager.estimator.end(); ++est) {
124  (*est)->reportStep(*this);
125  }
126  H5Awrite(stepAttrID, H5T_NATIVE_INT, &(++istep));
127  H5Fflush(fileID, H5F_SCOPE_LOCAL);
128  if (istep == nstep) {
129  H5Aclose(stepAttrID);
130  for (DataSetIter d=dataset.begin(); d!=dataset.end(); ++d) {
131  H5Dclose(*d);
132  }
133  H5Gclose(writingGroupID);
134  dataset.resize(0);
135  }
136 }

138 void H5ReportBuilder::reportScalarStep(const ScalarEstimator& est) {
139  hid_t space = H5Dget_space(*dset);
140  H5Sselect_none(space);
141 #if (H5_VERS_MAJOR>1)||((H5_VERS_MAJOR==1)&&(H5_VERS_MINOR>=8))||((H5_VERS_MAJOR==1)&&(H5_VERS_MINOR==6)&&(H5_VERS_RELEASE>=7))
142  hsize_t element = istep;
143  H5Sselect_elements(space, H5S_SELECT_SET, 1, &element);
144 #else
145  hsize_t element[][1] = {{istep}};
146  H5Sselect_elements(space, H5S_SELECT_SET, 1, (const hsize_t **)element);
147 #endif
148 
149  float a=est.getValue();
150  hsize_t dims=1, maxdims=1;
151  hid_t mspace = H5Screate_simple(1, &dims, &maxdims);
152  H5Dwrite(*dset, H5T_NATIVE_FLOAT, mspace, space, H5P_DEFAULT, &a);
153  dset++;
154 }

157  est.normalize();

169 void H5ReportBuilder::startArrayBlockedReport(const ArrayBlockedEstimator& est) {
170  hsize_t *dims;
171  dims = new hsize_t[est.getNDim()];
172  unsigned int maxDim=1, imaxDim=0, size=1;
173  // Find maximum dimension for compression.
174  for (int i=0; i<est.getNDim(); ++i) {
175  dims[i] = est.getExtent(i);
176  size *= dims[i];
177  if (dims[i]>maxDim) {maxDim=dims[i]; imaxDim=i;}
178  }
179  hid_t dataSpaceID = H5Screate_simple(est.getNDim(), dims, NULL);
180  bool useCompression = (size > 10000);
181  hid_t plist = H5P_DEFAULT;
182  if (useCompression) {
183  plist = H5Pcreate(H5P_DATASET_CREATE);
184  dims[imaxDim] = dims[imaxDim]*10000/size;
185  if (dims[imaxDim] == 0) dims[imaxDim]=1;
186  H5Pset_chunk(plist,est.getNDim(),dims);
187  H5Pset_deflate(plist,1);
188  }
189  delete dims;
190 #if (H5_VERS_MAJOR>1)||((H5_VERS_MAJOR==1)&&(H5_VERS_MINOR>=8))
191  hid_t dataSetID = H5Dcreate2(writingGroupID, est.getName().c_str(),
192  H5T_NATIVE_FLOAT, dataSpaceID, H5P_DEFAULT,
193  plist, H5P_DEFAULT);
194 #else
195  hid_t dataSetID = H5Dcreate(writingGroupID, est.getName().c_str(),
196  H5T_NATIVE_FLOAT, dataSpaceID, plist);
197 #endif
198  {
199  const std::string& typeString(est.getTypeString());
200  hsize_t dims=1;
201  hid_t dataSpaceID = H5Screate_simple(1, &dims, NULL);
202  hid_t strType = H5Tcopy(H5T_C_S1);
203  H5Tset_size(strType, typeString.length());
204 #if (H5_VERS_MAJOR>1)||((H5_VERS_MAJOR==1)&&(H5_VERS_MINOR>=8))
205  hid_t attrID = H5Acreate2(dataSetID, "type", strType, dataSpaceID,
206  H5P_DEFAULT, H5P_DEFAULT);
207 #else
208  hid_t attrID = H5Acreate(dataSetID, "type", strType, dataSpaceID,
209  H5P_DEFAULT);
210 #endif
211  H5Awrite(attrID, strType, typeString.c_str());
212  H5Sclose(dataSpaceID);
213  H5Aclose(attrID);
214  }
215  dataset.push_back(dataSetID);
216  if (est.hasError()) {
217 #if (H5_VERS_MAJOR>1)||((H5_VERS_MAJOR==1)&&(H5_VERS_MINOR>=8))
218  hid_t dataSetID = H5Dcreate2(writingGroupID, (est.getName()+"_err").c_str(),
219  H5T_NATIVE_FLOAT, dataSpaceID, H5P_DEFAULT,
220  plist, H5P_DEFAULT);
221 #else
222  hid_t dataSetID = H5Dcreate(writingGroupID, (est.getName()+"_err").c_str(),
223  H5T_NATIVE_FLOAT, dataSpaceID, plist);
224 #endif
225  dataset.push_back(dataSetID);
226  }
227  H5Sclose(dataSpaceID);
228  if (useCompression) H5Pclose(plist);
229  if (est.hasScale()) {
230  hsize_t dim=est.getNDim();
231  hid_t attrSpaceID = H5Screate_simple(1, &dim, NULL);
232 #if (H5_VERS_MAJOR>1)||((H5_VERS_MAJOR==1)&&(H5_VERS_MINOR>=8))
233  hid_t attrID = H5Acreate2(dataSetID, "scale", H5T_NATIVE_DOUBLE,
234  attrSpaceID, H5P_DEFAULT, H5P_DEFAULT);
235 #else
236  hid_t attrID = H5Acreate(dataSetID, "scale",
237  H5T_NATIVE_DOUBLE, attrSpaceID, H5P_DEFAULT);
238 #endif
239  H5Sclose(attrSpaceID);
240  H5Awrite(attrID, H5T_NATIVE_DOUBLE, est.getScale());
241  H5Aclose(attrID);
242  }
243 
244  if (est.hasOrigin()) {
245  hsize_t dim=est.getNDim();
246  hid_t attrSpaceID = H5Screate_simple(1, &dim, NULL);
247 #if (H5_VERS_MAJOR>1)||((H5_VERS_MAJOR==1)&&(H5_VERS_MINOR>=8))
248  hid_t attrID = H5Acreate2(dataSetID, "origin", H5T_NATIVE_DOUBLE,
249  attrSpaceID, H5P_DEFAULT, H5P_DEFAULT);
250 #else
251  hid_t attrID = H5Acreate(dataSetID, "origin",
252  H5T_NATIVE_DOUBLE, attrSpaceID, H5P_DEFAULT);
253 #endif
254  H5Sclose(attrSpaceID);
255  H5Awrite(attrID, H5T_NATIVE_DOUBLE, est.getOrigin());
256  H5Aclose(attrID);
257  }
258 
259  if (est.hasMin()) {
260  hsize_t dim=est.getNDim();
261  hid_t attrSpaceID = H5Screate_simple(1, &dim, NULL);
262 #if (H5_VERS_MAJOR>1)||((H5_VERS_MAJOR==1)&&(H5_VERS_MINOR>=8))
