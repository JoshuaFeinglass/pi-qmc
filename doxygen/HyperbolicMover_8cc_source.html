152 double HyperbolicMover::g(const double r, const double t,
153  const int ilevel) const {
154  double z = t / (2 * alpha) * sqrt(1.0 + 2 * mass * alpha * r * r / (t * t));
155  double k2 = gsl_sf_bessel_Kn(2, z);
156  return exp(t / (2 * alpha)) * t / (32 * pi * pi * alpha * z * z)
157  * pow(2 * mass / alpha, 1.5) * k2;

294 double HyperbolicMover::g1g2(const Vec center, const Vec delta,
295  const double teff, const int level) {

297  double r2 = sqrt(dot(center - delta, center - delta));
