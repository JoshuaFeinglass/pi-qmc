90 bool MultiLevelSampler::tryMove(double initialLnTranProb) {
91  double oldDeltaAction = initialLnTranProb;
92  for (int ilevel = nlevel; ilevel >= 0; --ilevel) {
93  if (accRejEst)
94  accRejEst->tryingMove(ilevel);
95 
96  // Make the trial move for this level A2B, and get the acc ratio and transition prob. A2B.
97  factor = defaultFactor;
98  double lnTranProb = mover.makeMove(*this, ilevel);
99  double deltaAction =
100  (action == 0) ? 0 : action->getActionDifference(*this, ilevel);
101  double piRatioBoA = -deltaAction + oldDeltaAction;
102  double acceptProb = exp(lnTranProb + piRatioBoA);
103 
104  // If you want to do DelayedRejection
105  if (delayedRejection && ilevel == nlevel - 1) {
106  if (RandomNumGenerator::getRand() > acceptProb) {
107  double accRatioA2B = acceptProb;
108  double transA2B = mover.getForwardProb();
109  for (int i = 0; i < nmoving; ++i) {
110  for (int islice = 0; islice < sectionBeads->getNSlice();
111  ++islice) {
112  (*rejectedBeads)(i, islice) = (*movingBeads)(i, islice);
113  }
114  }
115 
116  //Make another trial move for this level A2C, and get the acc ratio.
117  factor = newFactor;
118  double lnTranProb = mover.makeMove(*this, ilevel);
119  deltaAction =
120  (action == 0) ?
121  0 : action->getActionDifference(*this, ilevel);
122  double piRatioCoA = -deltaAction + oldDeltaAction;
123  double accRatioA2C = exp(lnTranProb + piRatioCoA);
124 
125  //Make the trial move to the rejected state C2B, and get the acc ratio and transition prob. C2B
126  factor = defaultFactor;
127  lnTranProb = mover.makeDelayedMove(*this, ilevel);
128  double transC2B = mover.getForwardProb();
129  double accRatioC2B = exp(lnTranProb - piRatioCoA + piRatioBoA);
130  accRatioC2B = (accRatioC2B >= 1) ? 1 : accRatioC2B;
131  double accRatioA2B2C = accRatioA2C * exp(transC2B + transA2B)
132  * (1 - accRatioC2B) / (1 - accRatioA2B);
133 
134  deltaAction = deltaAction - transC2B - transA2B
135  - log((1 - accRatioC2B) / (1 - accRatioA2B));
136  if (RandomNumGenerator::getRand() > accRatioA2B2C)
137  return false;
138  }
139  } else {
140  if (RandomNumGenerator::getRand() > acceptProb)
141  return false;
142  }
143  oldDeltaAction = deltaAction;
144 
145  if (accRejEst)
146  accRejEst->moveAccepted(ilevel);
147  }
148 
149  // Move accepted.
150  action->acceptLastMove();
151  // Put moved beads in section beads.
152  for (int islice = 0; islice < sectionBeads->getNSlice(); ++islice) {
153  movingBeads->copySlice(identityIndex, islice, *sectionBeads,
154  *movingIndex, islice);
155 
156  }
157  // Append the current permutation to section permutation.
158  const Permutation& perm(permutationChooser.getPermutation());
159  Permutation temp(perm);
160  for (int i = 0; i < nmoving; ++i) {
161  temp[i] = (*sectionPermutation)[particleChooser[i]];
162  }
163  for (int i = 0; i < nmoving; ++i) {
164  (*sectionPermutation)[particleChooser[i]] = temp[perm[i]];
165  }
166  return true;
167 }
