94 void PIMCParser::parse(const xmlXPathContextPtr& ctxt) {
95  // Read timestep and temperature.
96  xmlXPathObjectPtr obj = xmlXPathEval(BAD_CAST"//PIMC",ctxt);
97  xmlNodePtr& pimcNode=obj->nodesetval->nodeTab[0]; ctxt->node=pimcNode;
98  bool useDoublePaths=getBoolAttribute(pimcNode,"useDoublePaths");
99  xmlXPathFreeObject(obj);
100  double t=simInfo.getTemperature();
101  double tau=simInfo.getTau();
102  const int nslice=(int)(1.0/(tau*t)+0.5);
103  // Construct the Paths object.
104  if (mpi && (doubleAction || useDoublePaths) ) {
105  if (mpi->isMain()) std::cout << "Need double sampling." << std::endl;
106  }
107  if (mpi && mpi->getNWorker()>1) {
108  if (doubleAction) {
109  paths=new DoubleParallelPaths(simInfo.getNPart(),nslice,tau,
110  *simInfo.getSuperCell(),*mpi,beadFactory);
111  } else {
112  paths=new ParallelPaths(simInfo.getNPart(),nslice,tau,
113  *simInfo.getSuperCell(),*mpi,beadFactory);
114  }
115  } else {
116  paths=new SerialPaths(simInfo.getNPart(),
117  nslice,tau,*simInfo.getSuperCell(),beadFactory);
118  }
119  if (actionChoice) {
120  paths->setModelState(&actionChoice->getModelState());
121  }
122  // Parse the algorithm.
123  algorithm=parseAlgorithm(ctxt);
124 }
