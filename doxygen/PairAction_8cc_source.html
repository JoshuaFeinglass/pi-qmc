246  }

248  Vec prevDelta =sectionBeads(i,0)-sectionBeads(j,0);

297  double g0full = g0-g0X;

303  }

308  action=0.;
309  delta=sectionBeads(i,islice)-sectionBeads(j,islice);
310  cell.pbc(delta);
311  r=sqrt(dot(delta,delta));
312  q=0.5*(r+prevR);
313  if (level<3 && norder>0) {
314  Vec svec=delta-prevDelta; double s2=dot(svec,svec)/(q*q);
