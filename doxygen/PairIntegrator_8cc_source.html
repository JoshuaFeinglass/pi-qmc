122  double tauSave = tau;
123  tau *= scaleTau;
124  blitz::Range all(blitz::Range::all());
125  // Initialize potential grid.
126  for (int i=0; i<ngrid; ++i) {
127  double x2 = ((i-ngrid/2)*dr)*((i-ngrid/2)*dr);
128 #if NDIM>1
129  for (int j=0; j<ngrid; ++j) {
130  double y2 = ((j-ngrid/2)*dr)*((j-ngrid/2)*dr);
131 #if NDIM>2
132  for (int k=0; k<ngrid; ++k) {
133  double z2 = ((k-ngrid/2)*dr+q)*((k-ngrid/2)*dr+q);
134  vgrid(i,j,k) = pot(sqrt(x2+y2+z2));
135  }
136 #else
137  vgrid(i,j) = pot(sqrt(x2+y2));
138 #endif
139  }
140 #else
141  vgrid(i) = pot(sqrt(x2));
142 #endif
143  }
144  // Initialize wavefunction.
145  psi = 0.0; psi0 = 0.0;
146  double a = pow(dr,-NDIM);
147  int ndiff = ngrid/2/8/(norder>0?norder:1);
148  if (ndiff==0) ndiff=1;
149  IVecN1 index1 = ngrid/2;
150  IVecN2 index2 = ngrid/2;
151  int idata = 0;
152  for (int iorder=0; iorder<norder+1; ++iorder) {
153  index1(NDIM) = ngrid/2 + iorder*ndiff;
154  index2(NDIM+1) = ngrid/2 + iorder*ndiff;
155  for (int ioff=0; ioff < ((NDIM>1)?iorder+1:1); ++ioff) {
156  index1(0) = idata;
157  index2(1) = idata;
158 #if NDIM>1
159  index1(1) = ngrid/2 + ioff*ndiff;
160  index2(2) = ngrid/2 + ioff*ndiff;
161 #endif
162  for (int i=0; i<maxiter; ++i) {
163  index2(0) = i;
164  psi(index2) = a;
165  }
166  psi0(index1) = a;
167  ++idata;
168  }
169  }
170  //Propagate the wavefunction forward in a series of segments.
171  for (int iseg=0; iseg<nsegment; ++iseg) {
172  std::cout << "Segment " << iseg+1 << " (of " << nsegment
173  << ", q=" << q << ", tol=" << tol << "):" << std::endl;
174  propagate(tau/nsegment, tol);
175  }
176  // Calculate free propagator (includes FFT normalization).
177  double fftNorm = 1./ngridN;
178  fftw_complex *ptr0 = (fftw_complex*)psi0.data();
179  Array tFlat(tgrid.data(),IVec1(ngridN),blitz::neverDeleteData);
180  CArray2 psi0Flat(psi0.data(),IVec2(ndata,ngridN),blitz::neverDeleteData);
181  expFullTtau = exp(-tFlat*tau)*fftNorm;
182  fftw_execute_dft(fwd,ptr0,ptr0);
183  for (int idata=0; idata<ndata; ++idata) {
184  for (int igrid=0; igrid<ngridN; ++igrid) {
185  psi0Flat(idata,igrid) *= expFullTtau(igrid);
186  }
187  }
188  fftw_execute_dft(rev,ptr0,ptr0);
189  // Calculate arrays of G, G0, s2, and z.
190  idata = 0;
191  Vec r1=0., r2=0.;
192  index2(0) = 0;
193  for (int iorder=0; iorder<norder+1; ++iorder) {
194  r1(NDIM-1) = q + iorder*ndiff*dr;
195  r2(NDIM-1) = q - iorder*ndiff*dr;
196  index1(NDIM) = ngrid/2 - iorder*ndiff;
197  index2(NDIM+1) = ngrid/2 - iorder*ndiff;
198  for (int ioff=0; ioff < ((NDIM>1)?iorder+1:1); ++ioff) {
199 #if NDIM>1
200  r1(0) = ioff*ndiff*dr;
201  r2(0) = -ioff*ndiff*dr;
202  index1(1) = ngrid/2 - ioff*ndiff;
203  index2(2) = ngrid/2 - ioff*ndiff;
204 #endif

207  g(idata) = psi(index2).real();
208  g0(idata) = psi0(index1).real();

210  z(idata) = (sqrt(dot(r1,r1)) - sqrt(dot(r2,r2)))/q;
211  ++idata;
212  }
213  }
214  //Now find coefficients G(s^2)=sum c_ij s^2j z^2i
215  Array2 mat(ndata,ndata);
216  IArray ipiv(ndata);
217  u = log(g0/g);
218  int jdata = 0;
219  for (int jorder=0; jorder<norder+1; ++jorder) {
220  for (int joff=0; joff < ((NDIM>1)?jorder+1:1); ++joff) {
221  for (int i=0; i<ndata; ++i) {

223  }
224  ++jdata;
225  }
226  }
