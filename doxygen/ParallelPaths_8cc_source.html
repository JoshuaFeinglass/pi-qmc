132 void ParallelPaths::putBeads(int ifirstSlice, const Beads<NDIM>& inBeads,
133  const Permutation& inPermutation) const {
134  //First place the section beads back in the bead array.
135  int jfirstSlice=(ifirstSlice-ifirst+nslice)%nslice;
136  int nsectionSlice=inBeads.getNSlice();
137  for (int isectionSlice=0; isectionSlice<nsectionSlice; ++isectionSlice) {
138  int islice=isectionSlice+jfirstSlice;
139  for (int ipart=0; ipart<npart; ++ipart) {
140  beads(ipart,islice)=inBeads(ipart,isectionSlice);
141  }
142  }
143  // Now permute the following beads.
144  int jlastSlice=jfirstSlice+nsectionSlice;
145  if (!inPermutation.isIdentity()){
146  for (int islice=jlastSlice; islice<nprocSlice+2; ++islice) {
147  // Swap paths.
148  VArray buffer(npart);
149  for (int i=0; i<npart; ++i) buffer(i)=beads(inPermutation[i],islice);
150  for (int i=0; i<npart; ++i) beads(i,islice)=buffer(i);
151  }
152  permutation.prepend(inPermutation);
153  inversePermutation.setToInverse(permutation);
154  }
155 }

186 void ParallelPaths::shift(const int ishift) {
187 #ifdef ENABLE_MPI
188  // Transfer ishift beads between workers.
189  int idest=(iworker+nworker-1)%nworker;
190  int isrc=(iworker+1)%nworker;
191  double *sendbuf=&(beads(0,0)[0]);
192  double *recvbuf=&(buffer(0,nprocSlice-ishift)[0]);
193  mpi.getWorkerComm().Sendrecv(
194  sendbuf,(ishift+2)*NDIM*npart,MPI::DOUBLE,idest,1,
195  recvbuf,(ishift+2)*NDIM*npart,MPI::DOUBLE,isrc,1);
196  // Permute the ishift beads that we got from the next worker.
197  for (int islice=nprocSlice-ishift; islice<nprocSlice+2; ++islice) {
198  VArray vbuffer(npart);
199  for (int i=0; i<npart; ++i) vbuffer(i)=buffer(permutation[i],islice);
200  for (int i=0; i<npart; ++i) buffer(i,islice)=vbuffer(i);
201  }
202  // Shift the remaining beads that are in this worker.
203  for (int j=0; j<nprocSlice-ishift; ++j) {
204  for (int i=0; i<npart; ++i) {
205  buffer(i,j)=beads(i,j+ishift);
206  }
207  }
208  mpi.getWorkerComm().Barrier();
209  cycleArrays(buffer.getCoordArray(),beads.getCoordArray());
210 #endif
211 }

213 void ParallelPaths::setBuffers() {
214 #ifdef ENABLE_MPI
215  // Get the i=nprocSlice+1 buffer slice from i=1 slice on next worker.
216  int idest=(iworker+nworker-1)%nworker;
217  int isrc=(iworker+1)%nworker;
218  double *sendbuf=&(beads(0,1)[0]);
219  double *recvbuf=&(buffer(0,0)[0]);
220  mpi.getWorkerComm().Sendrecv(sendbuf,NDIM*npart,MPI::DOUBLE,idest,1,
221  recvbuf,NDIM*npart,MPI::DOUBLE,isrc,1);
222  // Permute the beads that we got from the next worker.
223  for (int i=0; i<npart; ++i) beads(i,nprocSlice+1)=buffer(permutation[i],0);
224  // Get the i=0 buffer slice from i=nprocSlice slice on next worker.
225  // First permute the beads that we will send to the next worker.
226  for (int i=0; i<npart; ++i) buffer(permutation[i],0)=beads(i,nprocSlice);
227  idest=(iworker+1)%nworker;
228  isrc=(iworker+nworker-1)%nworker;
229  sendbuf=&(buffer(0,0)[0]);
230  recvbuf=&(beads(0,0)[0]);
231  mpi.getWorkerComm().Sendrecv(sendbuf,NDIM*npart,MPI::DOUBLE,idest,1,
232  recvbuf,NDIM*npart,MPI::DOUBLE,isrc,1);
233 #endif
234 }
