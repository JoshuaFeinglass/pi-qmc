54  for (int islice=nStride; islice<nSlice-nStride; islice+=nStride) {
55  for (int iMoving=0; iMoving<nMoving; ++iMoving) {
56  const int i=index(iMoving);

60  cell.pbc(delta);

66  // Subtract action for old beads.
