35 double RingGateAction::getActionDifference(const SectionSamplerInterface& sampler, const int level) {
36  const Beads<NDIM>& sectionBeads=sampler.getSectionBeads();
37  const Beads<NDIM>& movingBeads=sampler.getMovingBeads();
38  const SuperCell& cell=sampler.getSuperCell();
39  const int nStride = 1 << level;
40  const int nSlice=sectionBeads.getNSlice();
41  const IArray& index=sampler.getMovingIndex();
42  const int nMoving=index.size();
43  double deltaAction=0;
44  double ktstride = tau*nStride;
45  double x = 0;
46  double y = 0;
47  double theta = 0;
48  for (int islice=nStride; islice<nSlice-nStride; islice+=nStride) {
49  for (int iMoving=0; iMoving<nMoving; ++iMoving) {
50  const int i=index(iMoving);
51  if (i<ifirst || i>=ifirst+npart) continue;
52  // Add action for moving beads.
53  Vec delta=movingBeads(iMoving,islice);
54  cell.pbc(delta);
55  x = delta[0];
56  y = delta[1];
57  theta = atan2(y,x);
58  deltaAction+=GVolt * (tanh(s * (theta + theta0)) - tanh(s * (theta - theta0))) / normalConst * ktstride;
59  // Subtract action for old beads.
60  delta = sectionBeads(i,islice);
61  cell.pbc(delta);
62  x = delta[0];
63  y = delta[1];
64  theta = atan2(y,x);
65  deltaAction-=GVolt * (tanh(s * (theta + theta0)) - tanh(s * (theta - theta0))) / normalConst * ktstride;
66  }
67  }
68  return deltaAction;
69 }
