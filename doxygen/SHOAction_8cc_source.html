25  const IArray& index=sampler.getMovingIndex();

31  double m=mass;
32  double div1=1.0/(2.0*sinhwt);
33  double div2=1.0/(2.0*tau*nStride);
34  for (int islice=nStride; islice<nSlice; islice+=nStride) {
35  for (int iMoving=0; iMoving<nMoving; ++iMoving) {
36  const int i=index(iMoving);
37  if (i<ifirst || i>=ifirst+npart) continue;
38  // Add action for moving beads.

40  Vec r0=movingBeads(iMoving,islice-nStride)-center;
41  Vec delta=r1-r0;

43  for (int idim=(NDIM-ndim); idim<NDIM; ++idim) {
44  r0r0+=r0[idim]*r0[idim];
45  r0r1+=r0[idim]*r1[idim];
46  r1r1+=r1[idim]*r1[idim];
47  delta2+=delta[idim]*delta[idim];
48  }

50  // Subtract action for old beads.

52  r0=sectionBeads(i,islice-nStride)-center;
53  delta=r1-r0;
