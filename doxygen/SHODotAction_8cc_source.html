41 #endif
42  for (int islice=nStride; islice<nSlice-nStride; islice+=nStride) {
43  int islicePrime = islice-nStride;
44  if( islicePrime<0 )
45  islicePrime+=nslice;
46  else if( islicePrime>=nslice )
47  islicePrime-=nslice;
48 
49  for (int iMoving=0; iMoving<nMoving; ++iMoving) {
50  const int i=index(iMoving);
51  if (i<ifirst || i>=ifirst+npart) continue;
52  // Add action for moving beads.
53  Vec position = movingBeads(iMoving,islice);
54  cell.pbc(position);
55 #if NDIM==3
56  deltaAction+=ktstride*(position[0]*position[0]+position[1]*position[1]);

63 #endif
