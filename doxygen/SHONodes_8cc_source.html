52 SHONodes::evaluate(const VArray &r1, const VArray &r2,
53  const int islice, bool scaleMagnitude) {
54  DetWithFlag result; result.err=false;
55  do { // Loop if scale is wrong to avoid overflow/underflow.
56  Matrix& mat(*matrix[islice]);
57  mat=0;
58  for(int jpart=0; jpart<npart; ++jpart) {
59  Vec rj=r1(jpart+ifirst);
60  double rj2=dot(rj,rj);
61  for(int ipart=0; ipart<npart; ++ipart) {
62  Vec ri=r2(ipart+ifirst);
63  double ri2=dot(ri,ri);
64  double rirj=dot(ri,rj);
65  mat(ipart,jpart)= scale*exp(-c*((ri2+rj2)*coshwt-2.0*rirj));
66  uarray(ipart,jpart) = -log(fabs(mat(ipart,jpart))+1e-100);
67  }
68  }
69  if (hasSpinModelState()) {
70  SpinModelState::IArray spin=(spinModelState->getSpinState());
71  for(int jpart=0; jpart<npart; ++jpart) {
72  for(int ipart=0; ipart<npart; ++ipart) {
73  if (spin(ipart) != spin(jpart)) mat(ipart,jpart) = 0.;
74  }
75  }
76  }
77  // Find dominant contribution to determinant (distroys uarray).
78  hungarian->solve(uarray.data());
79  for (int ipart=0; ipart<npart; ++ipart) {
80  kindex(islice,ipart) = (*hungarian)[ipart];
81  }
82  // Note: u(ipart,jpart=kindex(islice,ipart)) makes maximum contribution
83  // or lowest total action.
84 
85  // Calculate determinant and inverse.
86  int info=0;//LU decomposition
87  DGETRF_F77(&npart,&npart,mat.data(),&npart,ipiv.data(),&info);
88  if (info!=0) {
89  result.err = true;
90  std::cout << "BAD RETURN FROM ZGETRF!!!!" << std::endl;
91  nerror++;
92  if (nerror>1000) {
93  std::cout << "too many errors!!!!" << std::endl;
94  std::exit(-1);
95  }
96  }
97  double det = 1;
98  for (int i=0; i<npart; ++i) {
99  det*= mat(i,i);
100  det *= (i+1==ipiv(i))?1:-1;
101  }
102  DGETRI_F77(&npart,mat.data(),&npart,ipiv.data(),work.data(),&lwork,&info);
103  if (info!=0) {
104  result.err = true;
105  std::cout << "BAD RETURN FROM ZGETRI!!!!" << std::endl;
106  nerror++;
107  if (nerror>1000) {
108  std::cout << "too many errors!!!!" << std::endl;
109  std::exit(-1);
110  }
111  }
112  // watch for overflow or underflow.
113  if (scaleMagnitude && (!result.err
114  && (fabs(det)<1e-50 || fabs(det)>1e50) )) {
115  if (fabs(det)<1e-250) {
116  scale *= 2;
117  } else {
118  scale *= pow(fabs(det),-1./npart);
119  }
120  std::cout << "Slater determinant rescaled: scale = "
121  << scale << std::endl;
122  }
123  result.det = det;
124  }
125  while (scaleMagnitude && (result.err ||
126  (fabs(result.det)<1e-50 || fabs(result.det)>1e50)));
127  return result;
