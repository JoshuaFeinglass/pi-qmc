144 void SerialPaths::putBeads(int ifirstSlice, const Beads<NDIM>& inBeads,
145  const Permutation& inPermutation, const int ifirst,
146  const int nmslice) const {
147  //First place the section beads back in the bead array.
148  int nsectionSlice=nmslice;
149  for (int isectionSlice=0; isectionSlice<nsectionSlice; ++isectionSlice) {
150  int islice=isectionSlice+ifirstSlice+ifirst;
151  if (islice<0) {
152  // Do nothing, since this slice should not have been moved.
153  } else if (islice<nslice) {
154  inBeads.copySlice(isectionSlice+ifirst,beads,islice);
155  //for (int ipart=0; ipart<npart; ++ipart) {
156  // beads(ipart,islice)=inBeads(ipart,isectionSlice);
157  //}
158  } else {
159  islice%=nslice;
160  inBeads.copySlice(isectionSlice+ifirst,beads,permutation,islice);
161  //for (int ipart=0; ipart<npart; ++ipart) {
162  // beads(permutation[ipart],islice)=inBeads(ipart,isectionSlice);
163  //}
164  }
165  }
166  // Now permute the following beads.
167  if (!inPermutation.isIdentity()) {
168  int ifollowing=ifirstSlice+nsectionSlice;
169  Permutation temp(inPermutation);
170  if (ifollowing>nslice) {
171  ifollowing-=nslice;
172  temp.prepend(inversePermutation);
173  temp.append(permutation);
174  }
175 //BUG with aux beads!!!!
176  for (int islice=ifollowing; islice<nslice; ++islice) {
177  // Swap paths.
178  VArray buffer(npart);
179  for (int i=0; i<npart; ++i) buffer(i)=beads(temp[i],islice);
180  for (int i=0; i<npart; ++i) beads(i,islice)=buffer(i);
181  }
182  permutation.prepend(temp);
183  inversePermutation.setToInverse(permutation);
184  }
185 }

217 void SerialPaths::shift(const int ishift) {
218  if (ishift==0) return;
219  Permutation p(npart);
220  int nshift=(ishift>(buffer1.getNSlice())-1)?(buffer1.getNSlice()-1):ishift;
221  int remainder=ishift-nshift;
222  int islice=nslice-1;
223  getBeads(islice,buffer1);
224  for (islice-=nshift; islice>0; islice-=nshift) {
225  getBeads(islice,buffer2);
226  putBeads(islice,buffer1,p,1,nshift);
227  buffer1=buffer2;
228  }
229  putBeads(islice,buffer1,p,-islice,nshift+islice+1);
230  // Recursively repeat if more shifting needed.
231  if (remainder>0) shift(remainder);
232 }
