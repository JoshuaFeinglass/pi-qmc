236 double SmoothedGridPotential::getActionDifference(
237  const SectionSamplerInterface& sampler, const int ilevel) {
238  const Beads<NDIM>& sectionBeads=sampler.getSectionBeads();
239  const Beads<NDIM>& movingBeads=sampler.getMovingBeads();
240  const SuperCell& cell=sampler.getSuperCell();
241  const int nStride = 1 << ilevel;
242  const int nSlice=sectionBeads.getNSlice();
243  const IArray& index=sampler.getMovingIndex();
244  const int nMoving=index.size();
245  double deltaAction=0;
246  // Read values off of the grid.
247  for (int islice=0; islice<nSlice-nStride; islice+=nStride) {
248  for (int iMoving=0; iMoving<nMoving; ++iMoving) {
249  const int i=index(iMoving);
250  // Add action for moving beads. (Evaluate v at midpoint)
251  Vec r=movingBeads(iMoving,islice);
252  cell.pbc(r);
253  Vec delta=movingBeads(iMoving,islice+nStride);
254  delta-=r; cell.pbc(delta); delta*=0.5; r+=delta;
255  cell.pbc(r);
256  deltaAction+=v(r,i,ilevel)*tau*nStride;
257  // Subtract action for old beads.
258  r=sectionBeads(i,islice);
259  cell.pbc(r);
260  delta=sectionBeads(i,islice+nStride);
261  delta-=r; cell.pbc(delta); delta*=0.5; r+=delta;
262  cell.pbc(r);
263  deltaAction-=v(r,i,ilevel)*tau*nStride;
264  }
265  }
266  return deltaAction;
267 }
