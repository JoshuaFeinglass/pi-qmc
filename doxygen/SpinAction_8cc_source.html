27 #if NDIM==4

33  = *dynamic_cast<const Beads<4>*>(sampler.getSectionBeads().getAuxBeads(1));
34  const Beads<4>& movingBeads
35  = *dynamic_cast<const Beads<4>*>(sampler.getMovingBeads().getAuxBeads(1));
36 #endif
37  double wt=omega*tau*nStride;
38  double sinhwt=sinh(wt);
39  double coshwt=cosh(wt);
40  double m=mass;
41  double div1=1.0/(2.0*sinhwt);
42 //#endif
43  const int nSlice=sectionBeads.getNSlice();
44  const IArray& index=sampler.getMovingIndex();
45  const int nMoving=index.size();
46 
47  for (int islice=nStride; islice<nSlice; islice+=nStride) {
48  for (int iMoving=0; iMoving<nMoving; ++iMoving) {
49  const int i=index(iMoving);
50  //Add new action.
51  SVec s=movingBeads(iMoving,islice);
52  double dots=dot(s,s);
53  double f=invgc*exp(-gc*dots/l2);
54  diff-=nStride*tau*( bz*f*(s[0]*s[0]-s[1]*s[1]-s[2]*s[2]+s[3]*s[3])
55  + 2*bx*f*(-s[0]*s[2]+s[1]*s[3]) + 2*by*f*(s[0]*s[1]+s[2]*s[3]) )/l2;
56 //#if NDIM!=4
57  SVec s0=movingBeads(iMoving,islice-nStride);
58  diff+=m*omega*((dot(s,s)+dot(s0,s0))*coshwt-2.0*dot(s,s0))*div1;
59 //#endif
60 
61  //Subtract old action.
62  s=sectionBeads(i,islice);
63  dots=dot(s,s);
64  f=invgc*exp(-gc*dots/l2);
65  diff+=nStride*tau*(bz*f*( s[0]*s[0]-s[1]*s[1]-s[2]*s[2]+s[3]*s[3])
66  + 2*bx*f*(-s[0]*s[2]+s[1]*s[3]) + 2*by*f*(s[0]*s[1]+s[2]*s[3]) )/l2;
67 //#if NDIM!=4
68  s0=sectionBeads(i,islice-nStride);
69  diff-=m*omega*((dot(s,s)+dot(s0,s0))*coshwt-2.0*dot(s,s0))*div1;
70 //#endif
71  }
72  }
73  return diff;
74 }
