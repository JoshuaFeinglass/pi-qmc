56 double SpringAction::getActionDifference(const SectionSamplerInterface& sampler,
57  const int level) {
58  const Beads<NDIM>& sectionBeads=sampler.getSectionBeads();
59  const Beads<NDIM>& movingBeads=sampler.getMovingBeads();
60  const SuperCell& cell=sampler.getSuperCell();
61  const int nStride = 1 << level;
62  const int nSlice=sectionBeads.getNSlice();
63  const IArray& index=sampler.getMovingIndex();
64  const int nMoving=index.size();
65  double deltaAction=0;
66  for (int islice=nStride; islice<nSlice; islice+=nStride) {
67  for (int iMoving=0; iMoving<nMoving; ++iMoving) {
68  const int i=index(iMoving);
69  if (isStatic(i)) continue;
70  const int ispec=specIndex(i);
71  const double inv2Sigma2 = 0.25/(lambda(i)*tau*nStride);
72  // Add action for moving beads.
73  Vec delta=movingBeads.delta(iMoving,islice,-nStride);
74  cell.pbc(delta);
75  double temp = 1.0;
76  for (int idim=0;idim<NDIM;++idim) {
77  if (pg(level,ispec,idim)) {
78  temp *= pg(level,ispec,idim)->evaluate(delta[idim]);
79  } else {
80  deltaAction+=delta[idim]*delta[idim]*inv2Sigma2;
81  }
82  }
83  temp = 1.0/temp;
84  // Subtract action for old beads.
85  delta=sectionBeads.delta(i,islice,-nStride);
86  cell.pbc(delta);
87  for (int idim=0;idim<NDIM;++idim) {
88  if (pg(level,ispec,idim)) {
89  temp *= pg(level,ispec,idim)->evaluate(delta[idim]);
90  } else {
91  deltaAction-=delta[idim]*delta[idim]*inv2Sigma2;
92  }
93  }
94  deltaAction+=log(temp);
95  }
96  }
