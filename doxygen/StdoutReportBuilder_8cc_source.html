53 void StdoutReportBuilder::reportScalarStep(const ScalarEstimator& est) {
54  double value=est.getValue();
55  sum(iscalar)+=value; sum2(iscalar)+=value*value; norm(iscalar)+=1;
56  std::cout << est.getName();
57  if (est.getUnitName()!="") std::cout << " (" << est.getUnitName() << ")";
58  std::cout << ": " << value << ", "
59  << "Av=" << sum(iscalar)/(norm(iscalar))
60  << " +-" << sqrt(sum2(iscalar)-sum(iscalar)*sum(iscalar)
61  /(norm(iscalar)))/(norm(iscalar)-1)
62  << std::endl;
63  ++iscalar;
64 }
