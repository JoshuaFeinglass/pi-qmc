34 double StillWebAction::getActionDifference(const SectionSamplerInterface& sampler,
35  const int level) {
36  const Beads<NDIM>& sectionBeads=sampler.getSectionBeads();
37  const Beads<NDIM>& movingBeads=sampler.getMovingBeads();
38  const SuperCell& cell=sampler.getSuperCell();
39  const int nStride = 1 << level;
40  const int nSlice=sectionBeads.getNSlice();
41  const IArray& index=sampler.getMovingIndex();
42  const int nMoving=index.size();
43  double deltaAction=0;
44  double taueff = tau*nStride;
45  for (int islice=nStride; islice<nSlice-nStride; islice+=nStride) {
46  for (int iMoving=0; iMoving<nMoving; ++iMoving) {
47  const int i=index(iMoving);
48  for (int jpart=0; jpart<npart; ++jpart) {
49  if (jpart==i) continue;
50  Vec deltaj=movingBeads(iMoving,islice)-sectionBeads(jpart,islice);
51  cell.pbc(deltaj);
52  double rij=sqrt(dot(deltaj,deltaj));
53  deltaAction+=f2(rij,0,0)*taueff;
54  for (int kpart=0; kpart<jpart; ++kpart) {
55  if (kpart==i) continue;
56  // Add action for moving beads.
57  Vec deltak=movingBeads(iMoving,islice)-sectionBeads(kpart,islice);
58  cell.pbc(deltak);
59  double rik=sqrt(dot(deltak,deltak));
60  double costheta=dot(deltaj,deltak)/(rij*rik);
61  deltaAction+=h(rij,rik,costheta,0,0,0)*taueff;
62  }
63  deltaj=sectionBeads(i,islice)-sectionBeads(jpart,islice);
64  cell.pbc(deltaj);
65  rij=sqrt(dot(deltaj,deltaj));
66  deltaAction-=f2(rij,0,0)*taueff;
67  for (int kpart=0; kpart<jpart; ++kpart) {
68  if (kpart==i) continue;
69  // Subtract action for old beads.
70  Vec deltak=sectionBeads(i,islice)-sectionBeads(kpart,islice);
71  cell.pbc(deltak);
72  double rik=sqrt(dot(deltak,deltak));
73  double costheta=dot(deltaj,deltak)/(rij*rik);
74  deltaAction-=h(rij,rik,costheta,0,0,0)*taueff;
75  }
76  }
77  }
78  }
