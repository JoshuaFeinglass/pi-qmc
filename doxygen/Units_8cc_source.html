65 double Units::getLengthScaleIn(const std::string& unit, int factor) const {
66  // First remove power at end of unit, like 2 on nm2.
67  std::string unitbase = unit;
68  if (factor>1) {
69  if (factor != atoi(unitbase.substr(unitbase.length()-1).c_str())) {
70  std::cout << "WARNING :: power not right in unit, may want to throw error"
71  << std::endl;
72  }
73  unitbase = unitbase.substr(0,unitbase.length()-1);
74  }
75  unitMap::const_iterator i=lengthOut.find(unitbase);
76  if (i==lengthOut.end()) {
77  std::cout << "WARNING :: Length Unit not found, may want to throw error."
78  << std::endl; //Unit not found, may want to throw error.
79  return 1;
80  } else {
81  return 1.0/pow(i->second,factor);
82  }
83 }
