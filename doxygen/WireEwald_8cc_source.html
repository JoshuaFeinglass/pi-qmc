37 WireEwald::WireEwald(const SuperCell& cell, const double d, const double rmax,
38  const Vec fitWidth, const IVec gridSize)
39  : cell(cell), d(d), rmax(rmax), gridSize(gridSize),
40  dx(fitWidth[0] / double(gridSize[0]-1)),
41  dy(fitWidth[1] / double(gridSize[1]-1)),
42 #if NDIM==3
43  dz(fitWidth[2] / double(gridSize[2]-1)),
44 #endif
45  NN(3), MM(1)
46 #if NDIM==3
47  ,LL(1),
48 #endif
49 #if NDIM==2
50  ,polyCoeff((NN+1)*(MM+1)),
51  v(gridSize[0]*gridSize[1]),
52  x(gridSize[0]*gridSize[1]),
53  y(gridSize[0]*gridSize[1])
54 #endif
55 #if NDIM==3
56  polyCoeff((NN+1)*(MM+1)*(LL+1)),
57  v(gridSize[0]*gridSize[1]*gridSize[2]),
58  x(gridSize[0]*gridSize[1]*gridSize[2]),
59  y(gridSize[0]*gridSize[1]*gridSize[2]),
60  z(gridSize[0]*gridSize[1]*gridSize[2])
61 #endif
62  {
63  double L = cell.a[0];
64  int M = int(ceil(rmax / L));
65  v = 0.; x = 0.; y = 0.;
66  // Evaluate Coulomb interaction on the grid.
67  for (int i=0; i<gridSize[0]; ++i) {
68  for (int j=0; j<gridSize[1]; ++j) {
69 #if NDIM==2
70  x(i*gridSize[1]+j) = dx * i;
71  y(i*gridSize[1]+j) = dy * j;
72  for (int k=-M; k<M+1; ++k) {
73  if (k==0) continue;
74  v(i*gridSize[1]+j) += 2./sqrt(pow(dx * i + k * L,2) + pow(dy * j,2))
75  - 1./sqrt(pow(dx * i + k * L,2) + pow(dy * j + d,2))
76  - 1./sqrt(pow(dx * i + k * L,2) + pow(dy * j - d,2));
77  }
78 #endif
79 #if NDIM==3
80  for (int k=0; k<gridSize[2]; ++k) {
81  x(i*gridSize[1]*gridSize[2]+j*gridSize[2]+k) = dx * i;
82  y(i*gridSize[1]*gridSize[2]+j*gridSize[2]+k) = dy * j;
83  z(i*gridSize[1]*gridSize[2]+j*gridSize[2]+k) = dz * k;
84  for (int kk=-M; kk<M+1; ++kk) {
85  if (kk==0) continue;
86  v(i*gridSize[1]*gridSize[2]+j*gridSize[2]+k) +=
87  2./sqrt(pow(dx * i + k * L,2) + pow(dy * j,2) +pow(dz * k,2))
88  - 1./sqrt(pow(dx * i + k * L,2) + pow(dy * j + d,2)+pow(dz,2))
89  - 1./sqrt(pow(dx * i + k * L,2) + pow(dy * j - d,2)+pow(dz,2));
90  }
91  }
92 #endif
93  }
94  }
95  // Fit v to find the coefficients for the polynomial.
96  curveFit();
97 }

141 void WireEwald::curveFit() {
142 #if NDIM==2 or NDIM==3
143  // Define parameters
144  double tau = 1e-3;
145  double epsilon1 = 1e-15, epsilon2 = 1e-15, epsilon3 = 1e-15;
146  int KMAX = 100;
147 #if NDIM==2
148  Array1 p((NN+1)*(MM+1)), pnew((NN+1)*(MM+1));
149 #endif
150 #if NDIM==3
151  Array1 p((NN+1)*(MM+1)*(LL+1)), pnew((NN+1)*(MM+1)*(LL+1));
152 #endif
153  p = 1.;
154  // Evaluate the fitting function on the grid.
155  Array1 f(v.shape()[0]), fnew(v.shape()[0]);
156  f = 0.; fnew = 0.;
157  for (int i=0; i<gridSize[0]; ++i)
158  for (int j=0; j<gridSize[1]; ++j)
159 #if NDIM==2
160  f(i*gridSize[1]+j) = polynomial(NN,MM,x(i*gridSize[1]+j),y(i*gridSize[1]+j),p);
161 #endif
162 #if NDIM==3
163  for (int k=0; k<gridSize[2]; ++k)
164  f(i*gridSize[1]*gridSize[2]+j*gridSize[2]+k) =
165  polynomial(NN,MM,LL,x(i*gridSize[1]*gridSize[2]+j*gridSize[2]+k),
166  y(i*gridSize[1]*gridSize[2]+j*gridSize[2]+k),
167  z(i*gridSize[1]*gridSize[2]+j*gridSize[2]+k),
168  p);
169 #endif
170  // Calculate Jacobian on the grid.
171  DArray J(v.shape()[0],p.shape()[0]);
172  J = 0.;
173  for (int i=0; i<v.shape()[0]; ++i) {
174  for (int j=0; j<p.shape()[0]; ++j) {
175 #if NDIM==2
176  int k = floor(j/(MM+1.));
177  J(i,j) = pow(x(i),2*k) * pow(y(i),2*(j-k*(MM+1)));
178 #endif
179 #if NDIM==3
180  int k = int(floor(j/((MM+1.)*(LL+1.))));
181  int kk = int(floor(j/(LL+1.)));
182  J(i,j) = pow(x(i),2*k) * pow(y(i),2*(kk-k*(MM+1)))
183  * pow(z(i),2*(j-kk*(LL+1)));
184 #endif
185  }
186  }
187  firstIndex firstindex;
188  secondIndex secondindex;
189  thirdIndex thirdindex;
190  DArray A(J.shape()[1], J.shape()[1]);
191  A = sum(J(thirdindex, firstindex) * J(thirdindex, secondindex), thirdindex);
192  // Generate an identity matrix.
193  DArray identity(A.shape());
194  identity = 0.;
195  for (int i=0; i<A.shape()[0]; ++i)
196  identity(i,i) = 1.;
197  Array1 G(A.shape()[0]), epsilonP(v.shape());
198  epsilonP = v - f;
199  G = sum(J(secondindex, firstindex) * epsilonP(secondindex), secondindex);
200  int count = 0;
201  double nu = 2., mu = A(0,0);
202  for (int i=1; i<A.shape()[0]; ++i)
203  if (mu < A(i,i)) mu = A(i,i);
204  mu *= tau;
205  int dimMat = A.shape()[0];
206  int lwork = dimMat*dimMat;
207  Array1 work(lwork);
208  IArray ipiv(dimMat);
209  Array1 deltaP(dimMat);
210  DArray mat(A.shape());
211  double rhoe = 0.;
212  bool stop = max(abs(G)) < epsilon1;
213  while(!stop && (count < KMAX)) {
214  ++count;
215  do {
216  mat = A + mu * identity;
217  // LU decomposition
218  int info = 0;
219  DGETRF_F77(&dimMat, &dimMat, mat.data(), &dimMat, ipiv.data(), &info);
220  if (info!=0) {
221  std::cout<<"LU decomposition failed!"<<std::endl;
222  std::exit(-1);
223  }
224  DGETRI_F77(&dimMat, mat.data(), &dimMat, ipiv.data(), work.data(),
225  &lwork, &info);
226  if (info!=0) {
227  std::cout<<"Matrix inversion failed!"<<std::endl;
228  exit(-1);
229  }
230  deltaP = sum(mat(firstindex, secondindex) * G(secondindex), secondindex);
231  if (sum(deltaP*deltaP) <= epsilon2*epsilon2*sum(p*p))
232  stop = true;
233  else {
234  pnew = p + deltaP;
235  for (int i=0; i<gridSize[0]; ++i)
236  for (int j=0; j<gridSize[1]; ++j)
237 #if NDIM==2
238  fnew(i*gridSize[1]+j) = polynomial(NN,MM,x(i*gridSize[1]+j),y(i*gridSize[1]+j),pnew);
239 #endif
240 #if NDIM==3
241  for (int k=0; k<gridSize[2]; ++k)
242  fnew(i*gridSize[1]*gridSize[2]+j*gridSize[2]+k) =
243  polynomial(NN,MM,LL,x(i*gridSize[1]*gridSize[2]+j*gridSize[2]+k),
244  y(i*gridSize[1]*gridSize[2]+j*gridSize[2]+k),
245  z(i*gridSize[1]*gridSize[2]+j*gridSize[2]+k),
246  p);
247 #endif
248  rhoe = (sum(epsilonP*epsilonP) -
249  sum((v - fnew)*(v-fnew))) /
250  sum(deltaP * (mu* deltaP + G));
251  if (rhoe > 0) {
252  p = pnew;
253  f = fnew;
254  epsilonP = v - f;
255  G = sum(J(secondindex, firstindex) * (epsilonP(secondindex)), secondindex);
256  stop = ((max(abs(G)) < epsilon1) || (sum(epsilonP*epsilonP)) < epsilon3);
257  mu *=(0.3333333333333333333 > 1-pow(2*rhoe-1, 3))?0.3333333333333333333:
258  (1-pow(2*rhoe-1,3));
259  nu = 2.;
260  }
261  else {
262  mu *= nu;
263  nu *= 2.;
264  }
265  }
266  }while(!(rhoe > 0) && !stop);
267  }
268  polyCoeff = p;
269  std::cout<<polyCoeff<<std::endl;
270 #endif
271 }

274 inline double WireEwald::polynomial(int Nx, int My, double x, double y, const Array1& p) {
275  double f = 0.0;
276  for (int i=0; i<=Nx; ++i)
277  for(int j=0; j<=My; ++j)
278  f += p(i*(My+1)+j) * pow(x,2.*i) * pow(y,2.*j);
