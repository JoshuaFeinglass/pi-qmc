68 WireNodes::evaluate(const VArray &r1, const VArray &r2, const int islice,
69  bool scaleMagnitude) {
70  DetWithFlag result; result.err=false;
71  do { // Loop if scale is wrong to avoid overflow/underflow.
72  Matrix& mat(*matrix[islice]);
73  mat=0;
74  for(int jpart=0; jpart<npart; ++jpart) {
75  Vec rj=r1(jpart+ifirst)-center;
76  double rj2=0; for (int i=1; i<NDIM; ++i) rj2+=rj[i]*rj[i];
77  for(int ipart=0; ipart<npart; ++ipart) {
78  // Free particle in idim=0.
79  Vec delta(r1(jpart+ifirst)-r2(ipart+ifirst));
80  cell.pbc(delta);
81  double ear2 = pg->evaluate(delta[0]);
82  mat(ipart,jpart)=scale*ear2+1e-100;
83  // SHO in idim>0.
84  Vec ri=r2(ipart+ifirst)-center;
85  double ri2=0, rirj=0;
86  for (int i=1; i<NDIM; ++i) {
87  ri2+=ri[i]*ri[i];
88  rirj+=ri[i]*rj[i];
89  }
90  mat(ipart,jpart)*= exp(-c*((ri2+rj2)*coshwt-2.0*rirj));
91  uarray(ipart,jpart)=-log(fabs(mat(ipart,jpart))+1e-100);
92  }
93  }
94  if (hasSpinModelState()) {
95  SpinModelState::IArray spin=(spinModelState->getSpinState());
96  for(int jpart=0; jpart<npart; ++jpart) {
97  for(int ipart=0; ipart<npart; ++ipart) {
98  if (spin(ipart) != spin(jpart)) mat(ipart,jpart) = 0.;
99  }
100  }
101  }
102 
103  // Find dominant contribution to determinant (distroys uarray).
104  hungarian->solve(uarray.data());
105  for (int ipart=0; ipart<npart; ++ipart) {
106  kindex(islice,ipart) = (*hungarian)[ipart];
107  }
108  // Note: u(ipart,jpart=kindex(islice,ipart)) makes maximum contribution
109  // or lowest total action.
110  // Next calculate determinant and inverse of slater matrix.
111  int info=0;//LU decomposition
112  DGETRF_F77(&npart,&npart,mat.data(),&npart,ipiv.data(),&info);
113  if (info!=0) {
114  result.err = true;
115  std::cout << "BAD RETURN FROM ZGETRF!!!!" << std::endl;
116  nerror++;
117  if (nerror>1000) {
118  std::cout << "too many errors!!!!" << std::endl;
119  std::exit(-1);
120  }
121  }
122  double det = 1;
123  for (int i=0; i<npart; ++i) {
124  det*= mat(i,i);
125  det *= (i+1==ipiv(i))?1:-1;
126  }
127  DGETRI_F77(&npart,mat.data(),&npart,ipiv.data(),work.data(),&lwork,&info);
128  if (info!=0) {
129  result.err = true;
130  std::cout << "BAD RETURN FROM ZGETRI!!!!" << std::endl;
131  nerror++;
132  if (nerror>1000) {
133  std::cout << "too many errors!!!!" << std::endl;
134  std::exit(-1);
135  }
136  }
137  // watch for overflow or underflow.
138  if (scaleMagnitude && (!result.err
139  && (fabs(det)<1e-50 || fabs(det)>1e50) )) {
140  if (fabs(det)<1e-250) {
141  scale *= 2;
142  } else {
143  scale *= pow(fabs(det),-1./npart);
144  }
145  std::cout << "Slater determinant rescaled: scale = "
146  << scale << std::endl;
147  }
148  result.det = det;
149  }
150  while (scaleMagnitude && (result.err ||
151  (fabs(result.det)<1e-50 || fabs(result.det)>1e50)));
152  return result;
153 }

155 void WireNodes::evaluateDotDistance(const VArray &r1, const VArray &r2,
156  const int islice, Array &d1, Array &d2) {
157  PeriodicGaussian* pgSave;
158  pgSave=pg;
159  double tauSave=tau;
160  // Calculate gradient with finite differences.
161  // First calculate distance at smaller beta.
162  tau = tauSave*(1.-EPSILON);
163  pg=pgm;
164  evaluate(r1, r2, islice, false);
165  evaluateDistance(r1, r2, islice, temp1, temp2);
166  // Next calculate distance at larger beta.
167  tau = tauSave*(1.+EPSILON);
168  pg=pgp;
169  evaluate(r1, r2, islice, false);
170  evaluateDistance(r1, r2, islice, d1, d2);
171  // Now use finite differences.
172  double denom=1./(2*tau*EPSILON);
173  d1 = denom*(d1-temp1);
174  d2 = denom*(d2-temp2);
175  // Restore pg and tau to proper temperature.
176  pg=pgSave;
177  tau=tauSave;
178 }

278  isNewMatrixUpdated=false;

315  for (int k=0; k<npart; ++k) {
316  (*bvec[islice])(ipart,jmoving)
