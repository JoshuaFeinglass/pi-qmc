84 double XMLUnitParser::getFieldStrengthAttribute(

88  if (unit!="") {

90  if( loc != std::string::npos ) {
91  std::string eunit = unit.substr(0,loc);
92  std::string lunit = unit.substr(loc+1);

101 void XMLUnitParser::parseUnitAndValue(const xmlNodePtr &node,
102  const std::string &attName) {
